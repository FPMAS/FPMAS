68 template<typename T>
69 T MutexClient<T>::read(DistributedId id, int location) {
70 FPMAS_LOGD(this->comm.getRank(), "MUTEX_CLIENT", "reading node %s from %i", FPMAS_C_STR(id), location);
71 // Starts non-blocking synchronous send
72 fpmas::api::communication::Request req;
73 this->id_mpi.Issend(id, location, server_pack.getEpoch() | Tag::READ, req);
74
75 // Keep responding to other READ / ACQUIRE request to avoid deadlock,
76 // until the request has been received
77 server_pack.waitSendRequest(req);
78
79 // The request has been received : it is assumed that the receiving proc is
80 // now responding so we can safely wait for response without deadlocking
81 // TODO : this is not thread safe.
82 fpmas::api::communication::Status read_response_status;
83 server_pack.waitResponse(data_mpi, location, Tag::READ_RESPONSE, read_response_status);
84
85 return data_mpi.recv(read_response_status.source, read_response_status.tag);
86 }
