93  void LocationManager<T>::updateLocations(
94  const NodeMap& local_nodes_to_update
95  ) {
96  FPMAS_LOGD(comm.getRank(), "LOCATION_MANAGER", "Updating node locations...");
97  // Useful types
98  typedef
99  std::unordered_map<int, std::vector<DistributedId>>
100  DistributedIdMap;
101  typedef
102  std::unordered_map<int, std::vector<std::pair<DistributedId, int>>>
103  LocationMap;
104 
105  /*
106  * Step 1 : send updated locations to origins, and receive
107  * locations of this origin's nodes
108  */
109  DistributedIdMap exported_updated_locations;
110  for(auto node : local_nodes_to_update) {
111  // Updates local node field
112  node.second->setLocation(comm.getRank());
113 
114  if(node.first.rank() != comm.getRank()) {
115  // Notify node origin that node is currently on this proc
116  exported_updated_locations[node.first.rank()]
117  .push_back(node.first);
118  } else {
119  // No need to export the new node location to itself :
120  // just locally set this node location to this proc
121  managed_nodes_locations[node.first]
122  = comm.getRank();
123  }
124  }
125 
126  // Export / import location updates of managed_nodes_locations
127  DistributedIdMap imported_updated_locations =
128  id_mpi.migrate(exported_updated_locations);
129  // Updates the managed_nodes_locations data
130  for(auto list : imported_updated_locations) {
131  for(auto node_id : list.second) {
132  managed_nodes_locations[node_id]
133  = list.first;
134  }
135  }
136 
137  // If some distant node has this proc has origin, the current
138  // location has already been updated and won't be requested to this
139  // proc himself
140  for(auto node : distant_nodes) {
141  if(node.first.rank() == comm.getRank()) {
142  node.second->setLocation(
143  managed_nodes_locations[node.first]
144  );
145  }
146  }
147 
148  /*
149  * Step 2 : ask for current locations of distant_nodes
150  */
151  DistributedIdMap location_requests;
152  for(auto node : distant_nodes) {
153  if(node.first.rank() != comm.getRank()) {
154  // For distant_nodes that has an origin different from this
155  // proc, asks for the current location of node to its
156  // origin proc
157  location_requests[node.first.rank()]
158  .push_back(node.first);
159  }
160  }
161  // Export / import requests
162  DistributedIdMap imported_location_requests
163  = id_mpi.migrate(location_requests);
164 
165  // Builds requests response
166  LocationMap exported_locations;
167  for(auto list : imported_location_requests) {
168  for(auto node_id : list.second) {
169  // for each requested node location, respond a tuple
170  // {node_id, currentLocation} to the requesting proc
171  exported_locations[list.first]
172  .push_back({node_id, managed_nodes_locations[node_id]});
173  }
174  }
175 
176  // Import / export responses
177  LocationMap imported_locations
178  = location_mpi.migrate(exported_locations);
179 
180  // Finally, updates the distant_nodes locations from the requests
181  // responses
182  for(auto list : imported_locations) {
183  for(auto location : list.second) {
184  distant_nodes.at(location.first)->setLocation(location.second);
185  }
186  }
187  FPMAS_LOGD(comm.getRank(), "LOCATION_MANAGER", "Node locations updated.");
188  }
189 }
