112 void MutexServer<T>::handleIncomingReadAcquireLock() {
113 fpmas::api::communication::Status status;
114
115 // Check read request
116 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::READ, status)) {
117 DistributedId id = id_mpi.recv(status.source, status.tag);
118 FPMAS_LOGD(this->comm.getRank(), "MUTEX_SERVER", "receive read request %s from %i", FPMAS_C_STR(id), status.source);
119 this->handleRead(id, status.source);
120 }
121
122 // Check acquire request
123 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::ACQUIRE, status)) {
124 DistributedId id = id_mpi.recv(status.source, status.tag);
125 FPMAS_LOGD(this->comm.getRank(), "MUTEX_SERVER", "receive acquire request %s from %i", FPMAS_C_STR(id), status.source);
126 this->handleAcquire(id, status.source);
127 }
128
129 // Check lock
130 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::LOCK, status)) {
131 DistributedId id = id_mpi.recv(status.source, status.tag);
132 FPMAS_LOGD(this->comm.getRank(), "MUTEX_SERVER", "receive lock request %s from %i", FPMAS_C_STR(id), status.source);
133 this->handleLock(id, status.source);
134 }
135
136 // Check shared lock
137 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::LOCK_SHARED, status)) {
138 DistributedId id = id_mpi.recv(status.source, status.tag);
139 FPMAS_LOGD(this->comm.getRank(), "MUTEX_SERVER", "receive shared lock request %s from %i", FPMAS_C_STR(id), status.source);
140 this->handleLockShared(id, status.source);
141 }

189 * Handles a read request.
190 * The request is transmitted to the corresponding ReaderWriter instance, that
191 * will respond immediately if the resource is available or put the request in an waiting
192 * queue otherwise.

194 template<typename T>
195 void MutexServer<T>::handleRead(DistributedId id, int source) {
196 auto* mutex = mutex_map.at(id);
197 if(mutex->locked()) {
198 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "Enqueueing READ request of node %s for %i",
199 FPMAS_C_STR(id), source);
200 mutex->pushRequest(Request(id, source, MutexRequestType::READ));
201 } else {
202 respondToRead(id, source);
203 }
204 }

250 template<typename T>
251 void MutexServer<T>::respondToRequests(HardSyncMutex* mutex) {
252 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "Unqueueing requests...", "");
253 std::queue<Request> requests = mutex->requestsToProcess();
254 while(!requests.empty()) {
255 Request request = requests.front();
256 switch(request.type) {
257 case MutexRequestType::READ :
258 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->READ(%s)",
259 request.source, FPMAS_C_STR(request.id));
260 respondToRead(request.id, request.source);
261 break;
262 case MutexRequestType::LOCK :
263 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->LOCK(%s)",
264 request.source, FPMAS_C_STR(request.id));
265 respondToLock(request.id, request.source);
266 break;
267 case MutexRequestType::ACQUIRE :
268 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->ACQUIRE(%s)",
269 request.source, FPMAS_C_STR(request.id));
270 respondToAcquire(request.id, request.source);
271 break;
272 case MutexRequestType::LOCK_SHARED :
273 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->LOCK_SHARED(%s)",
274 request.source, FPMAS_C_STR(request.id));
275 respondToLockShared(request.id, request.source);
276 }
277 requests.pop();
278 }

361 bool MutexServer<T>::respondToRequests(HardSyncMutex* mutex, const Request& request_to_wait) {
362 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "Unqueueing requests...", "");
363 std::queue<Request> requests = mutex->requestsToProcess();
364 while(!requests.empty()) {
365 Request request = requests.front();
366 if(request.source != Request::LOCAL) {
367 switch(request.type) {
368 case MutexRequestType::READ :
369 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->READ(%s)",
370 request.source, FPMAS_C_STR(request.id));
371 respondToRead(request.id, request.source);
372 break;
373 case MutexRequestType::LOCK :
374 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->LOCK(%s)",
375 request.source, FPMAS_C_STR(request.id));
376 respondToLock(request.id, request.source);
377 break;
378 case MutexRequestType::ACQUIRE :
379 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->ACQUIRE(%s)",
380 request.source, FPMAS_C_STR(request.id));
381 respondToAcquire(request.id, request.source);
382 break;
383 case MutexRequestType::LOCK_SHARED :
384 FPMAS_LOGV(comm.getRank(), "MUTEX_SERVER", "%i->LOCK_SHARED(%s)",
385 request.source, FPMAS_C_STR(request.id));
386 respondToLockShared(request.id, request.source);
387 }
388 requests.pop();
389 } else {
390 if(request == request_to_wait) {
391 return true;
392 }
393 }
394 }
395 return false;
396 }
