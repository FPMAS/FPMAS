194 static void synchronize(api::model::SpatialModel<GraphCellType>& model) {
195 for(auto cell : model.cellGroup().localAgents()) {
196 auto graph_cell = dynamic_cast<ReachableCell*>(cell);

198 for(auto neighbor : cell->node()->outNeighbors(api::model::CELL_SUCCESSOR))
