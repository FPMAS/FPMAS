195 static void synchronize(api::model::SpatialModel<GraphCellType>& model) {
196 for(auto cell : model.cellGroup().localAgents()) {
197 auto graph_cell = dynamic_cast<ReachableCell*>(cell);
198 graph_cell->clearReachableCells();
199 for(auto neighbor : cell->node()->outNeighbors(api::model::CELL_SUCCESSOR))
200 graph_cell->addReachableCell(neighbor->getId());
201 }
202 model.graph().synchronize();
