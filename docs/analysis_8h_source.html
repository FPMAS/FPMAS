52 template<typename T>
53 std::unordered_map<DistributedId, std::vector<DistributedId>>
54 distant_nodes_outgoing_neighbors(
55 fpmas::api::graph::DistributedGraph<T>& graph,
56 fpmas::api::graph::LayerId layer) {
57 // MPI instance used to exchange DistributedIds requests
58 communication::TypedMpi<std::vector<DistributedId>> requests_mpi(
59 graph.getMpiCommunicator());
60 // Builds a request for each \DISTANT node in the graph
61 std::unordered_map<int, std::vector<DistributedId>> requests;
62 for(auto node : graph.getLocationManager().getDistantNodes())
63 requests[node.second->location()].push_back(node.first);
64 requests = requests_mpi.allToAll(requests);
65
66 // MPI instance used to send request results. Results are received
67 // as a vector of pairs {requested node ids, {list of ids of
68 // outgoing neighbors}}.
69 communication::TypedMpi<std::vector<std::pair<DistributedId, std::vector<DistributedId>>>> results_mpi(graph.getMpiCommunicator());
70 std::unordered_map<int, std::vector<std::pair<DistributedId, std::vector<DistributedId>>>> results;
71 for(auto request : requests) {
72 for(auto node_id : request.second) {
73 // Safely gets neighbors of a LOCAL node
74 auto edges = graph.getNode(node_id)->getOutgoingEdges(layer);
75 // Stores neighbors ids in a list
76 std::vector<DistributedId> neighbors(edges.size());
77 auto it = neighbors.begin();
78 for(auto edge : edges)
79 *it++ = edge->getTargetNode()->getId();
80 // Save the {node_id, {neighbors}} pair, ready to be
81 // transmitted
82 results[request.first].push_back({node_id, neighbors});
83 }
84 }
85 // Exchange results
86 results = results_mpi.allToAll(results);
87
88 // Gathers results from all processes
89 std::unordered_map<DistributedId, std::vector<DistributedId>>
90 distant_nodes_neighbors;
91 for(auto item : results)
92 for(auto neighbors : item.second)
93 distant_nodes_neighbors[neighbors.first] = neighbors.second;
94 // Returns results for all DISTANT nodes on the current process
95 return distant_nodes_neighbors;
96 }

123 template<typename T>
124 double clustering_coefficient(
125 fpmas::api::graph::DistributedGraph<T>& graph,
126 fpmas::api::graph::LayerId layer) {
127 double c = 0;
128 auto distant_nodes_neighbors = distant_nodes_outgoing_neighbors(graph, layer);
129
130 for(auto node : graph.getLocationManager().getLocalNodes()) {
131 std::size_t n = 0;
132
133 // Neighbors of `node`= the set of incoming AND outgoing
134 // neighbors, without duplicate. The purpose of the algorithm is
135 // then to count edges between those neighbors.
136 std::set<DistributedId> neighbors;
137 for(auto edge : node.second->getIncomingEdges(layer)) {
138 neighbors.insert(edge->getSourceNode()->getId());
139 }
140 for(auto edge : node.second->getOutgoingEdges(layer)) {
141 neighbors.insert(edge->getTargetNode()->getId());
142 }
143 // Do not consider self edges
144 neighbors.erase(node.first);
145
146 for(auto node_id : neighbors) {
147 // The objective is to count edges between neighbor with
148 // others in the neighbors list. Only OUTGOING edges are
149 // considered: since each neighbors is considered, this
150 // guarantees that each edge is counted exactly once.
151 auto neighbor = graph.getNode(node_id);
152 if(neighbor->state() == fpmas::api::graph::LOCAL) {
153 // Edges can be retrieved locally
154 for(auto edge : neighbor->getOutgoingEdges(layer)) {
155 auto id = edge->getTargetNode()->getId();
156 if(
157 // Do not consider edge to the considered node
158 id != node.first
159 // Only edges from a node's neighbor to an other
160 && neighbors.find(id) != neighbors.end()) {
161 ++n;
162 }
163 }
164 } else {
165 // The neighbor is \DISTANT, but its neighbors can be
166 // retrieved with the distant_nodes_neighbors ids map.
167 for(auto id : distant_nodes_neighbors[node_id])
168 if(id != node.first && neighbors.find(id) != neighbors.end()) {
169 ++n;
170 }
171 }
172 }
173 std::size_t k = neighbors.size();
174 if(k > 1)
175 c += (double) n / (k*(k-1));
176 }
177 communication::TypedMpi<std::size_t> int_mpi(graph.getMpiCommunicator());
178 communication::TypedMpi<double> double_mpi(graph.getMpiCommunicator());
179 c = communication::all_reduce(double_mpi, c);
180 std::size_t total_node_count =
181 communication::all_reduce(int_mpi, graph.getLocationManager().getLocalNodes().size());
182 if(total_node_count > 0)
183 return c/total_node_count;
184 else
185 return 0;
186 }
