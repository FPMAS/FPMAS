494  template<typename T, typename IdType, int N> void Graph<T, IdType, N>::removeNode(IdType nodeId) {
495  FPMAS_LOGD(-1, "GRAPH", "Removing node %lu", nodeId);
496  node_ptr node_to_remove;
497  try {
498  node_to_remove = nodes.at(nodeId);
499  } catch(std::out_of_range) {
500  throw exceptions::node_out_of_graph(nodeId);
501  }
502 
503  // Deletes incoming arcs
504  for(auto& layer : node_to_remove->layers) {
505  for(auto arc : layer.getIncomingArcs()) {
506  try {
507  FPMAS_LOGV(
508  -1,
509  "GRAPH",
510  "Unlink incoming arc %lu",
511  arc->getId()
512  );
513  this->unlink(arc);
514  } catch (exceptions::arc_out_of_graph<IdType> e) {
515  FPMAS_LOGE(-1, "GRAPH", "Error unlinking incoming arc : %s", e.what());
516  throw;
517  }
518  }
519  // Deletes outgoing arcs
520  for(auto arc : layer.getOutgoingArcs()) {
521  try {
522  FPMAS_LOGV(
523  -1,
524  "GRAPH",
525  "Unlink outgoing arc %lu",
526  arc->getId()
527  );
528  this->unlink(arc);
529  } catch (exceptions::arc_out_of_graph<IdType> e) {
530  FPMAS_LOGE(-1, "GRAPH", "Error unlinking outgoing arc : %s", e.what());
531  throw;
532  }
533  }
534  }
535 
536  // Removes the node from the global nodes index
537  nodes.erase(nodeId);
538  // Deletes the node
539  delete node_to_remove;
540  FPMAS_LOGD(-1, "GRAPH", "Node %lu removed.", nodeId);
541  }
