286  template<class T> bool Graph<T>::unlink(Arc<T>* arc) {
287  // Removes the incoming arcs from the incoming/outgoing
288  // arc lists of target/source nodes.
289  this->removeArc(arc, &arc->getSourceNode()->outgoingArcs);
290  this->removeArc(arc, &arc->getTargetNode()->incomingArcs);
291 
292  // Removes the arc from the global arcs index
293  if(this->arcs.erase(arc->getId()) > 0) {
294  // Deletes the arc only if it was in the standard arcs set.
295  // Arcs can actually have a different nature not handled by
296  // this base class (e.g. : ghost arcs), and in consequence
297  // such special arcs must be managed by corresponding
298  // sub-classes (e.g. : DistributedGraph).
299  delete arc;
300  return true;
301  }
302  return false;
