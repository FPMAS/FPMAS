294  template<NODE_PARAMS> bool Graph<NODE_PARAMS_SPEC>::unlink(ARC* arc) {
295  // Removes the incoming arcs from the incoming/outgoing
296  // arc lists of target/source nodes.
297  this->removeArc(arc, &arc->getSourceNode()->layer(arc->layer).outgoingArcs);
298  this->removeArc(arc, &arc->getTargetNode()->layer(arc->layer).incomingArcs);
299 
300  // Removes the arc from the global arcs index
301  if(this->arcs.erase(arc->getId()) > 0) {
302  // Deletes the arc only if it was in the standard arcs set.
303  // Arcs can actually have a different nature not handled by
304  // this base class (e.g. : ghost arcs), and in consequence
305  // such special arcs must be managed by corresponding
306  // sub-classes (e.g. : DistributedGraph).
307  delete arc;
308  return true;
309  }
310  return false;
311  }
