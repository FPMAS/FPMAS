302  template<typename T, int N> void Graph<T, N>::removeNode(NodeId nodeId) {
303  FPMAS_LOGD(-1, "GRAPH", "Removing node %lu", nodeId);
304  Node<T, N>* node_to_remove;
305  try {
306  node_to_remove = nodes.at(nodeId);
307  } catch(std::out_of_range) {
308  throw exceptions::node_out_of_graph(nodeId);
309  }
310 
311  // Deletes incoming arcs
312  for(auto& layer : node_to_remove->layers) {
313  for(auto arc : layer.getIncomingArcs()) {
314  try {
315  FPMAS_LOGV(
316  -1,
317  "GRAPH",
318  "Unlink incoming arc %lu",
319  arc->getId()
320  );
321  this->unlink(arc);
322  } catch (exceptions::arc_out_of_graph e) {
323  FPMAS_LOGE(-1, "GRAPH", "Error unlinking incoming arc : %s", e.what());
324  throw;
325  }
326  }
327  // Deletes outgoing arcs
328  for(auto arc : layer.getOutgoingArcs()) {
329  try {
330  FPMAS_LOGV(
331  -1,
332  "GRAPH",
333  "Unlink outgoing arc %lu",
334  arc->getId()
335  );
336  this->unlink(arc);
337  } catch (exceptions::arc_out_of_graph e) {
338  FPMAS_LOGE(-1, "GRAPH", "Error unlinking outgoing arc : %s", e.what());
339  throw;
340  }
341  }
342  }
343 
344  // Removes the node from the global nodes index
345  nodes.erase(nodeId);
346  // Deletes the node
347  delete node_to_remove;
348  FPMAS_LOGD(-1, "GRAPH", "Node %lu removed.", nodeId);
349  }
