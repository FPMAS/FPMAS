431  template<class T> void Graph<T>::removeNode(unsigned long node_id) {
432  Node<T>* node_to_remove = nodes.at(node_id);
433 
434  // Deletes incoming arcs
435  for(auto arc : node_to_remove->getIncomingArcs()) {
436  Node<T>* source_node = arc->getSourceNode();
437  std::vector<Arc<T>*>* out_arcs = &source_node->outgoingArcs;
438 
439  // Removes the incoming arcs from the outgoingArcs list of the
440  // associated source node
441  for(auto it = out_arcs->begin(); it != out_arcs->end();) {
442  if((*it)->getId() == arc->getId()) {
443  out_arcs->erase(it);
444  }
445  else {
446  it++;
447  }
448  }
449  // Removes the arc from the global arcs index
450  this->arcs.erase(arc->getId());
451  // Deletes the arc
452  delete arc;
453  }
454 
455  // Deletes outgoing arcs
456  for(auto arc : node_to_remove->getOutgoingArcs()) {
457  Node<T>* target_node = arc->getTargetNode();
458  std::vector<Arc<T>*>* in_arcs = &target_node->incomingArcs;
459 
460  // Removes the outgoing arcs from the incomingArcs list of the
461  // associated target node
462  for(auto it = in_arcs->begin(); it != in_arcs->end();) {
463  if((*it)->getId() == arc->getId()) {
464  in_arcs->erase(it);
465  }
466  else {
467  it++;
468  }
469  }
470  // Removes the arc from the global arcs index
471  this->arcs.erase(arc->getId());
472  // Deletes the arc
