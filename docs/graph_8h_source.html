296  template<typename T, int N> bool Graph<T, N>::unlink(Arc<T, N>* arc) {
297  // Removes the incoming arcs from the incoming/outgoing
298  // arc lists of target/source nodes.
299  this->removeArc(arc, &arc->getSourceNode()->layer(arc->layer).outgoingArcs);
300  this->removeArc(arc, &arc->getTargetNode()->layer(arc->layer).incomingArcs);
301 
302  // Removes the arc from the global arcs index
303  if(this->arcs.erase(arc->getId()) > 0) {
304  // Deletes the arc only if it was in the standard arcs set.
305  // Arcs can actually have a different nature not handled by
306  // this base class (e.g. : ghost arcs), and in consequence
307  // such special arcs must be managed by corresponding
308  // sub-classes (e.g. : DistributedGraph).
309  delete arc;
310  return true;
311  }
312  return false;
313  }
