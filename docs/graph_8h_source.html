447  template<class T> void Graph<T>::removeNode(unsigned long node_id) {
448  Node<T>* node_to_remove = nodes.at(node_id);
449 
450  // Deletes incoming arcs
451  for(auto arc : node_to_remove->getIncomingArcs()) {
452  Node<T>* source_node = arc->getSourceNode();
453  std::vector<Arc<T>*>* out_arcs = &source_node->outgoingArcs;
454 
455  // Removes the incoming arcs from the outgoingArcs list of the
456  // associated source node
457  for(auto it = out_arcs->begin(); it != out_arcs->end();) {
458  if(*it == arc) {
459  out_arcs->erase(it);
460  it = out_arcs->end();
461  }
462  else {
463  it++;
464  }
465  }
466  // Removes the arc from the global arcs index
467  this->arcs.erase(arc->getId());
468  // Deletes the arc
469  delete arc;
470  }
471 
472  // Deletes outgoing arcs
473  for(auto arc : node_to_remove->getOutgoingArcs()) {
474  Node<T>* target_node = arc->getTargetNode();
475  std::vector<Arc<T>*>* in_arcs = &target_node->incomingArcs;
476 
477  // Removes the outgoing arcs from the incomingArcs list of the
478  // associated target node
479  for(auto it = in_arcs->begin(); it != in_arcs->end();) {
480  if(*it == arc) {
481  in_arcs->erase(it);
482  it = in_arcs->end();
483  }
484  else {
485  it++;
486  }
487  }
488  // Removes the arc from the global arcs index
489  this->arcs.erase(arc->getId());
490  // Deletes the arc
491  delete arc;
492  }
