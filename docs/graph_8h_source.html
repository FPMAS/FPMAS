273  template<class T> bool Graph<T>::unlink(Arc<T>* arc) {
274  // Removes the incoming arcs from the incoming/outgoing
275  // arc lists of target/source nodes.
276  this->removeArc(arc, &arc->getSourceNode()->outgoingArcs);
277  this->removeArc(arc, &arc->getTargetNode()->incomingArcs);
278 
279  // Removes the arc from the global arcs index
280  if(this->arcs.erase(arc->getId()) > 0) {
281  // Deletes the arc only if it was in the standard arcs set.
282  // Arcs can actually have a different nature not handled by
283  // this base class (e.g. : ghost arcs), and in consequence
284  // such special arcs must be managed by corresponding
285  // sub-classes (e.g. : DistributedGraph).
286  delete arc;
287  return true;
288  }
