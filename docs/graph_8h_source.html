268  template<class T> bool Graph<T>::unlink(Arc<T>* arc) {
269  // Removes the incoming arcs from the incoming/outgoing
270  // arc lists of target/source nodes.
271  this->removeArc(arc, &arc->getSourceNode()->outgoingArcs);
272  this->removeArc(arc, &arc->getTargetNode()->incomingArcs);
273 
274  // Removes the arc from the global arcs index
275  if(this->arcs.erase(arc->getId()) > 0) {
276  // Deletes the arc only if it was in the standard arcs set.
277  // Arcs can actually have a different nature not handled by
278  // this base class (e.g. : ghost arcs), and in consequence
279  // such special arcs must be managed by corresponding
280  // sub-classes (e.g. : DistributedGraph).
281  delete arc;
282  return true;
283  }
284  return false;
