247  template<class T, template<typename> class S> void unpack_obj_multi_fn(
248  void *data,
249  int num_gid_entries,
250  int num_ids,
251  ZOLTAN_ID_PTR global_ids,
252  int *sizes,
253  int *idx,
254  char *buf,
255  int *ierr) {
256 
257  DistributedGraph<T, S>* graph = (DistributedGraph<T, S>*) data;
258 
259  // The same arc can be imported multiple times from
260  // different procs.
261  // e.g : When nodes 1 and 0 are linked and live on different
262  // procs, and they are both imported to an other proc.
263  // This set records which arcs are imported and is used to
264  // ignore already imported ones.
265  // TODO: this might probably be optimized, removing
266  // duplicates when import / export maps are computed
267  std::set<unsigned long> receivedArcIds;
268 
269  for (int i = 0; i < num_ids; ++i) {
270  json json_arc = json::parse(&buf[idx[i]]);
271 
272  // Unserialize and process the arc only if it has not
273  // been imported already.
274  if(receivedArcIds.count(json_arc.at("id").get<unsigned long>()) == 0) {
275 
276  // Json is unserialized in a temporary arc, with "fake"
277  // nodes that just contains ID. We don't know yet which
278  // nodes are on this local process or not.
279  Arc<SyncDataPtr<T>> tempArc = json_arc.get<Arc<SyncDataPtr<T>>>();
280 
281  receivedArcIds.insert(tempArc.getId());
282 
283  unsigned long sourceId = tempArc.getSourceNode()->getId();
284  bool sourceNodeIsLocal = graph->getNodes().count(
285  sourceId
286  ) > 0;
287 
288  unsigned long targetId = tempArc.getTargetNode()->getId();
289  bool targetNodeIsLocal = graph->getNodes().count(
290  targetId
291  ) > 0;
292 
293  if(!sourceNodeIsLocal || !targetNodeIsLocal) {
294  // Source or target node is distant : a GhostNode must be used.
295  if(sourceNodeIsLocal) {
296  // The source node of the received arc is
297  // contained in the local graph, so the target
298  // node is distant
299  GhostNode<T, S>* ghost;
300  if(graph->getGhost().getNodes().count(targetId) == 0) {
301  // No ghost node as been created yet for
302  // this node (from an other arc imported at
303  // this period, or from an other period)
304  ghost = graph->getGhost().buildNode(targetId);
305 
306  // Updates the proxy from info received
307  // with arc info, to know where the target
308  // node can be found
309  std::array<int, 2> targetLocation = json_arc.at("target").get<std::array<int, 2>>();
310  graph->getProxy().setOrigin(targetId, targetLocation[0]);
311  graph->getProxy().setCurrentLocation(targetId, targetLocation[1]);
312  }
313  else {
314  // Use the existing GhostNode
315  ghost = graph->getGhost().getNodes().at(targetId);
316  }
317  // Links the ghost node with the local node
318  // using a GhostArc
319  graph->getGhost().link(graph->getNode(sourceId), ghost, tempArc.getId());
320  }
321  else if (targetNodeIsLocal) {
322  // The target node of the received arc is
323  // contained in the local graph, so the source
324  // node is distant
325 
326  // Same process has above
327  GhostNode<T, S>* ghost;
328  if(graph->getGhost().getNodes().count(sourceId) == 0) {
329  ghost = graph->getGhost().buildNode(sourceId);
330 
331  std::array<int, 2> sourceLocation = json_arc.at("source").get<std::array<int, 2>>();
332  graph->getProxy().setOrigin(sourceId, sourceLocation[0]);
333  graph->getProxy().setCurrentLocation(sourceId, sourceLocation[1]);
334  }
335  else {
336  ghost = graph->getGhost().getNodes().at(sourceId);
337  }
338  graph->getGhost().link(ghost, graph->getNode(targetId), tempArc.getId());
339  }
340  else {
341  throw std::runtime_error(
342  "Invalid arc received : [" + std::to_string(sourceId) + "," + std::to_string(targetId) + "]"
343  );
344 
345  }
346  }
347  else {
348  // Both nodes are local, no ghost needs to be used.
349  // tempArc source and targetNodes are automatically
350  // deleted by this function
351  graph->link(
352  sourceId,
353  targetId,
354  tempArc.getId()
355  );
356  }
357  delete tempArc.getSourceNode();
358  delete tempArc.getTargetNode();
359  }
360  }

370  template<class T, template<typename> class S> void post_migrate_pp_fn_olz(
371  void *data,
372  int num_gid_entries,
373  int num_lid_entries,
374  int num_import,
375  ZOLTAN_ID_PTR import_global_ids,

379  int num_export,
380  ZOLTAN_ID_PTR export_global_ids,
381  ZOLTAN_ID_PTR export_local_ids,
382  int *export_procs,
383  int *export_to_part,
384  int *ierr) {
385 
386  // The next steps will remove exported nodes from the local
387  // graph, creating corresponding ghost nodes when necessary
388  DistributedGraph<T, S>* graph = (DistributedGraph<T, S>*) data;
389 
390  // Computes the set of ids of exported nodes
391  std::set<unsigned long> exportedNodeIds;
392  for(int i = 0; i < graph->export_node_num; i++) {
393  exportedNodeIds.insert(zoltan::utils::read_zoltan_id(
394  &graph->export_node_global_ids[i * num_gid_entries])
395  );
396  }
397 
398  // Builds ghost nodes when necessary.
399  // For each exported node, a ghost node is created if and only
400  // if at least one local node is still connected to the
401  // exported node.
402  for(auto id : exportedNodeIds) {
403  Node<SyncDataPtr<T>>* node = graph->getNode(id);
404  bool buildGhost = false;
405  for(auto arc : node->getOutgoingArcs()) {
406  if(exportedNodeIds.count(arc->getTargetNode()->getId()) == 0) {
407  buildGhost = true;
408  break;
409  }
410  }
411  if(!buildGhost) {
412  for(auto arc : node->getIncomingArcs()) {
413  if(exportedNodeIds.count(arc->getSourceNode()->getId()) == 0) {
414  buildGhost = true;
415  break;
416  }
417  }
418  }
419  if(buildGhost) {
420  graph->getGhost().buildNode(*node, exportedNodeIds);
421  }
422  }
423 
424  // Remove nodes and collect fossils
425  FossilArcs<SyncDataPtr<T>> ghostFossils;
426  for(auto id : exportedNodeIds) {
427  ghostFossils.merge(graph->removeNode(id));
428  }
429 
430  // TODO : schema with fossils example
431  // For info, see the Fossil and removeNode docs
432  graph->getGhost().clear(ghostFossils);
433  }
