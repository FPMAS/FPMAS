211  bool targetNodeIsLocal = graph->getNodes().count(
212  targetId
213  ) > 0;

216  // Source or target node is distant : a GhostNode must be used.
217  if(sourceNodeIsLocal) {
218  // The source node of the received arc is
219  // contained in the local graph, so the target
220  // node is distant
221  GhostNode<T>* ghost;
222  if(graph->getGhost()->getNodes().count(targetId) == 0) {
223  // No ghost node as been created yet for
224  // this node (from an other arc imported at
225  // this period, or from an other period)
226  ghost = graph->getGhost()->buildNode(targetId);
227 
228  // Updates the proxy from info received
229  // with arc info, to know where the target
230  // node can be found
231  std::array<int, 2> targetLocation = json_arc.at("target").get<std::array<int, 2>>();
232  graph->getProxy()->setOrigin(targetId, targetLocation[0]);
233  graph->getProxy()->setCurrentLocation(targetId, targetLocation[1]);
234  }
235  else {
236  // Use the existing GhostNode
237  ghost = graph->getGhost()->getNodes().at(targetId);
238  }

240  // using a GhostArc
241  graph->getGhost()->link(graph->getNode(sourceId), ghost, tempArc.getId());
242  }

245  // contained in the local graph, so the source
246  // node is distant
247

250  if(graph->getGhost()->getNodes().count(sourceId) == 0) {
251  ghost = graph->getGhost()->buildNode(sourceId);
252
