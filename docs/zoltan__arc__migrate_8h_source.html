247  template<typename T, int N, SYNC_MODE> void unpack_obj_multi_fn(
248  void *data,
249  int num_gid_entries,
250  int num_ids,
251  ZOLTAN_ID_PTR global_ids,
252  int *sizes,
253  int *idx,
254  char *buf,
255  int *ierr) {
256 
257  DistributedGraph<T, S, N>* graph = (DistributedGraph<T, S, N>*) data;
258  FPMAS_LOGV(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "unpack_obj_multi_fn");
259 
260  // The same arc can be imported multiple times from
261  // different procs.
262  // e.g : When nodes 1 and 0 are linked and live on different
263  // procs, and they are both imported to an other proc.
264  // This set records which arcs are imported and is used to
265  // ignore already imported ones.
266  // TODO: this might probably be optimized, removing
267  // duplicates when import / export maps are computed
268  std::set<ArcId> receivedArcIds;
269 
270  for (int i = 0; i < num_ids; ++i) {
271  json json_arc = json::parse(&buf[idx[i]]);
272 
273  // Unserialize and process the arc only if it has not
274  // been imported already.
275  if(receivedArcIds.count(json_arc.at("id").get<ArcId>()) == 0) {
276 
277  // Json is unserialized in a temporary arc, with "fake"
278  // nodes that just contains ID. We don't know yet which
279  // nodes are on this local process or not.
280  Arc<std::unique_ptr<SyncData<T,N,S>>, N> tempArc = json_arc.get<Arc<std::unique_ptr<SyncData<T,N,S>>, N>>();
281 
282  receivedArcIds.insert(tempArc.getId());
283 
284  NodeId sourceId = tempArc.getSourceNode()->getId();
285  bool sourceNodeIsLocal = graph->getNodes().count(
286  sourceId
287  ) > 0;
288 
289  NodeId targetId = tempArc.getTargetNode()->getId();
290  bool targetNodeIsLocal = graph->getNodes().count(
291  targetId
292  ) > 0;
293 
294  if(!sourceNodeIsLocal || !targetNodeIsLocal) {
295  // Source or target node is distant : a GhostNode must be used.
296  if(sourceNodeIsLocal) {
297  // The source node of the received arc is
298  // contained in the local graph, so the target
299  // node is distant
300  GhostNode<T, N, S>* ghost;
301  if(graph->getGhost().getNodes().count(targetId) == 0) {
302  // No ghost node as been created yet for
303  // this node (from an other arc imported at
304  // this period, or from an other period)
305  ghost = graph->getGhost().buildNode(targetId);
306 
307  // Updates the proxy from info received
308  // with arc info, to know where the target
309  // node can be found
310  std::array<int, 2> targetLocation = json_arc.at("target").get<std::array<int, 2>>();
311  graph->getProxy().setOrigin(targetId, targetLocation[0]);
312  graph->getProxy().setCurrentLocation(targetId, targetLocation[1]);
313  }
314  else {
315  // Use the existing GhostNode
316  ghost = graph->getGhost().getNodes().at(targetId);
317  }
318  // Links the ghost node with the local node
319  // using a GhostArc
320  graph->getGhost().link(graph->getNode(sourceId), ghost, tempArc.getId(), tempArc.layer);
321  }
322  else if (targetNodeIsLocal) {
323  // The target node of the received arc is
324  // contained in the local graph, so the source
325  // node is distant
326 
327  // Same process has above
328  GhostNode<T, N, S>* ghost;
329  if(graph->getGhost().getNodes().count(sourceId) == 0) {
330  ghost = graph->getGhost().buildNode(sourceId);
331 
332  std::array<int, 2> sourceLocation = json_arc.at("source").get<std::array<int, 2>>();
333  graph->getProxy().setOrigin(sourceId, sourceLocation[0]);
334  graph->getProxy().setCurrentLocation(sourceId, sourceLocation[1]);
335  }
336  else {
337  ghost = graph->getGhost().getNodes().at(sourceId);
338  }
339  graph->getGhost().link(ghost, graph->getNode(targetId), tempArc.getId(), tempArc.layer);
340  }
341  else {
342  throw std::runtime_error(
343  "Invalid arc received : [" + std::to_string(sourceId) + "," + std::to_string(targetId) + "]"
344  );
345 
346  }
347  }
348  else {
349  // Both nodes are local, no ghost needs to be used.
350  graph->link(
351  sourceId,
352  targetId,
353  tempArc.getId(),
354  tempArc.layer
355  );
356  }
357  delete tempArc.getSourceNode();
358  delete tempArc.getTargetNode();
359  }
360  }
361  FPMAS_LOGV(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "unpack_obj_multi_fn : done");
362  }

371  template<typename T, int N, SYNC_MODE> void post_migrate_pp_fn_olz(
372  void *data,
373  int num_gid_entries,
374  int num_lid_entries,
375  int num_import,
376  ZOLTAN_ID_PTR import_global_ids,
377  ZOLTAN_ID_PTR import_local_ids,
378  int *import_procs,
379  int *import_to_part,
380  int num_export,
381  ZOLTAN_ID_PTR export_global_ids,
382  ZOLTAN_ID_PTR export_local_ids,
383  int *export_procs,
384  int *export_to_part,
385  int *ierr) {
386 
387  // The next steps will remove exported nodes from the local
388  // graph, creating corresponding ghost nodes when necessary
389  DistributedGraph<T, S, N>* graph = (DistributedGraph<T, S, N>*) data;
390  FPMAS_LOGV(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "post_migrate_pp_fn_olz");
391 
392  // Computes the set of ids of exported nodes
393  std::set<NodeId> exportedNodeIds;
394  for(int i = 0; i < graph->export_node_num; i++) {
395  exportedNodeIds.insert(zoltan::utils::read_zoltan_id(
396  &graph->export_node_global_ids[i * num_gid_entries])
397  );
398  }
399 
400  // Builds ghost nodes when necessary.
401  // For each exported node, a ghost node is created if and only
402  // if at least one local node is still connected to the
403  // exported node.
404  for(auto id : exportedNodeIds) {
405  Node<std::unique_ptr<SyncData<T,N,S>>, N>* node = graph->getNode(id);
406  bool buildGhost = false;
407  for(auto layer : node->getLayers()) {
408  for(auto arc : layer.getOutgoingArcs()) {
409  if(exportedNodeIds.count(arc->getTargetNode()->getId()) == 0) {
410  buildGhost = true;
411  break;
412  }
413  }
414  }
415  if(!buildGhost) {
416  for(auto layer : node->getLayers()) {
417  for(auto arc : layer.getIncomingArcs()) {
418  if(exportedNodeIds.count(arc->getSourceNode()->getId()) == 0) {
419  buildGhost = true;
420  break;
421  }
422  }
423  }
424  }
425  if(buildGhost) {
426  FPMAS_LOGD(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "Building ghost node %lu", node->getId());
427  graph->getGhost().buildNode(*node, exportedNodeIds);
428  }
429  }
430 
431  // Remove nodes
432  for(auto id : exportedNodeIds) {
433  FPMAS_LOGD(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "Removing exported node %lu", id);
434  graph->removeNode(id);
435  }
436 
437  FPMAS_LOGV(graph->getMpiCommunicator().getRank(), "ZOLTAN_ARC", "post_migrate_pp_fn_olz : done");
438  }
