91  // Note : Zoltan should be able to directly use unsigned long as ids.
92  // However, this require to set up compile time flags to Zoltan, so we
93  // use 2 entries of the default zoltan unsigned int instead, to make it
94  // compatible with any Zoltan installation.
95  // Passing compile flags to the custom embedded CMake Zoltan
96  // installation might also be possible.
97  unsigned long node_id(const ZOLTAN_ID_PTR global_ids) {
98  return ((unsigned long) (global_ids[0] << 16)) | global_ids[1];
99  }
