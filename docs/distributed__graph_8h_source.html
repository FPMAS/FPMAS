455  template<class T, SYNC_MODE, int N>
456  Arc<std::unique_ptr<SyncData<T,N,S>>, N>* DistributedGraph<T, S, N>
457  ::link(NodeId source, NodeId target, ArcId arcId, LayerId layerId) {
458  if(this->getNodes().count(source) > 0) {
459  // Source is local
460  Node<std::unique_ptr<SyncData<T,N,S>>, N>* sourceNode
461  = this->getNode(source);
462  if(this->getNodes().count(target) > 0) {
463  // Source and Target are local, completely local operation
464  Node<std::unique_ptr<SyncData<T,N,S>>, N>* targetNode
465  = this->getNode(target);
466  return this->Graph<std::unique_ptr<SyncData<T,N,S>>, N>::link(
467  sourceNode, targetNode, arcId, layerId
468  );
469  } else {
470  // Target is distant
471  GhostNode<T, N, S>* targetNode
472  = this->getGhost().getNode(target);
473  syncMode.initLink(source, target, arcId, layerId);
474  // link local -> distant
475  auto arc = this->getGhost().link(
476  sourceNode, targetNode, arcId, layerId
477  );
478  syncMode.notifyLinked(arc);
479  return arc;
480  }
481  } else {
482  // Source is distant
483  if(this->getNodes().count(target) > 0) {
484  // Source is local
485  GhostNode<T, N, S>* sourceNode
486  = this->getGhost().getNode(source);
487  Node<std::unique_ptr<SyncData<T,N,S>>, N>* targetNode
488  = this->getNode(target);
489  syncMode.initLink(source, target, arcId, layerId);
490  // link distant -> local
491  auto arc = this->getGhost().link(
492  sourceNode, targetNode, arcId, layerId
493  );
494  syncMode.notifyLinked(arc);
495  return arc;
496  }
497  else {
498  // Source and target are distant
499  // Allowed only if the two nodes are ghosts on this graph
500  // TODO: better exception handling
501  GhostNode<T, N, S>* sourceNode
502  = this->getGhost().getNode(source);
503  GhostNode<T, N, S>* targetNode
504  = this->getGhost().getNode(target);
505 
506  syncMode.initLink(source, target, arcId, layerId);
507  // link distant -> distant
508  auto arc = this->getGhost().link(
509  sourceNode, targetNode, arcId, layerId
510  );
511  syncMode.notifyLinked(arc);
512  return arc;
513  }
514  }

518  template<typename T, SYNC_MODE, int N> void DistributedGraph<T, S, N>::unlink(
519  ArcId arcId
520  ) {
521  if(this->getArcs().count(arcId) > 0) {
522  FPMAS_LOGD(this->mpiCommunicator.getRank(), "DIST_GRAPH", "Unlinking local arc %lu", arcId);
523  // Arc is local
524  this->Graph<std::unique_ptr<SyncData<T,N,S>>, N>::unlink(arcId);
525  } else {
526  FPMAS_LOGD(this->mpiCommunicator.getRank(), "DIST_GRAPH", "Unlinking ghost arc %lu", arcId);
527  Arc<std::unique_ptr<SyncData<T,N,S>>, N>* arc;
528  try {
529  arc = this->getGhost().getArcs().at(arcId);
530  } catch (std::out_of_range) {
531  FPMAS_LOGE(
532  this->mpiCommunicator.getRank(),
533  "DIST_GRAPH", "%s",
534  FPMAS::graph::base::exceptions::arc_out_of_graph(arcId).what()
535  );
536  }
537  NodeId source = arc->getSourceNode()->getId();
538  NodeId target = arc->getTargetNode()->getId();
539  ArcId arcId = arc->getId();
540  LayerId layer = arc->layer;
541 
542  this->syncMode.initUnlink(arc);
543  arc->unlink();
544  if(this->getNodes().count(source) == 0) {
545  this->getGhost().clear((GhostNode<T,N,S>*) arc->getSourceNode());
546  }
547  if(this->getNodes().count(target) == 0) {
548  this->getGhost().clear((GhostNode<T,N,S>*) arc->getTargetNode());
549  }
550  this->getGhost().deleteArc((GhostArc<T,N,S>*) arc);
551  this->syncMode.notifyUnlinked(source, target, arcId, layer);
552  }
553  FPMAS_LOGD(this->mpiCommunicator.getRank(), "DIST_GRAPH", "Unlinked arc %lu", arcId);
554  }

697  template<class T, SYNC_MODE, int N> void DistributedGraph<T, S, N>::distribute(
698  std::unordered_map<NodeId, std::pair<int, int>> partition
699  ) {
700 
701  // Import lists
702  int num_import = 0;
703  ZOLTAN_ID_PTR import_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
704  int* import_procs = (int*) std::malloc(0);
705 
706  // Export lists
707  this->export_node_num = 0;
708  this->export_node_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
709  this->export_node_procs = (int*) std::malloc(0);
710 
711  for(auto item : partition) {
712  if(item.second.first == this->getMpiCommunicator().getRank()) {
713  if(item.second.second != this->getMpiCommunicator().getRank()) {
714  this->export_node_num++;
715  this->export_node_global_ids = (ZOLTAN_ID_PTR)
716  std::realloc(this->export_node_global_ids, 2 * this->export_node_num * sizeof(ZOLTAN_ID_TYPE));
717 
718  write_zoltan_id(item.first, &this->export_node_global_ids[2*(this->export_node_num-1)]);
719  this->export_node_procs = (int*)
720  std::realloc(this->export_node_procs, this->export_node_num * sizeof(int));
721  this->export_node_procs[this->export_node_num-1] = item.second.second;
722  }
723 
724  } else {
725  if(item.second.second == this->getMpiCommunicator().getRank()) {
726  num_import++;
727  import_global_ids = (ZOLTAN_ID_PTR)
728  std::realloc(import_global_ids, 2 * num_import * sizeof(ZOLTAN_ID_TYPE));
729  write_zoltan_id(item.first, &import_global_ids[2*(num_import-1)]);
730  import_procs = (int*)
731  std::realloc(import_procs, num_import * sizeof(int));
732  import_procs[num_import-1] = item.second.first;
733 
734  }
735 
736  }
737  }
738  int* import_to_part = (int*) std::malloc(num_import * sizeof(int));
739  for (int i = 0; i < num_import; ++i) {
740  import_to_part[i] = import_procs[i];
