374  // nodes.

376 
377  // Unused buffer
378  ZOLTAN_ID_TYPE export_arcs_local_ids[0];
379 
380  // Arcs to import
381  this->zoltan.Migrate(
382  -1,
383  NULL,
384  NULL,
385  NULL,
386  NULL,
387  this->export_arcs_num,
388  this->export_arcs_global_ids,
389  export_arcs_local_ids,
390  this->export_arcs_procs,
391  this->export_arcs_procs // parts = procs
392  );
393  std::free(this->export_arcs_global_ids);
394  std::free(this->export_arcs_procs);
395  }
396 
397  // Frees the zoltan buffers
398  this->zoltan.LB_Free_Part(
399  &import_global_ids,
400  &import_local_ids,
401  &import_procs,
402  &import_to_part
403  );
404 
405  this->zoltan.LB_Free_Part(
406  &this->export_node_global_ids,
407  &export_local_ids,
408  &this->export_node_procs,
409  &export_to_part
410  );
411

413  this->proxy.synchronize();
414 
415  // When called for the first time, PARTITION is used, and so

417  this->zoltan.Set_Param("LB_APPROACH", "REPARTITION");

435  template<class T, SYNC_MODE, typename LayerType, int N> void DistributedGraph<T, S, LayerType, N>::distribute(
436  std::unordered_map<unsigned long, std::pair<int, int>> partition
437  ) {
438 
439  // Import lists
440  int num_import = 0;
441  ZOLTAN_ID_PTR import_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
442  int* import_procs = (int*) std::malloc(0);
443 
444  // Export lists
445  this->export_node_num = 0;
446  this->export_node_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
447  this->export_node_procs = (int*) std::malloc(0);
448 
449  for(auto item : partition) {
450  if(item.second.first == this->getMpiCommunicator().getRank()) {
451  if(item.second.second != this->getMpiCommunicator().getRank()) {
452  this->export_node_num++;
453  this->export_node_global_ids = (ZOLTAN_ID_PTR)
454  std::realloc(this->export_node_global_ids, 2 * this->export_node_num * sizeof(ZOLTAN_ID_TYPE));
455 
456  write_zoltan_id(item.first, &this->export_node_global_ids[2*(this->export_node_num-1)]);
457  this->export_node_procs = (int*)
458  std::realloc(this->export_node_procs, this->export_node_num * sizeof(int));
459  this->export_node_procs[this->export_node_num-1] = item.second.second;
460  }
461 
462  } else {
463  if(item.second.second == this->getMpiCommunicator().getRank()) {
464  num_import++;
465  import_global_ids = (ZOLTAN_ID_PTR)
466  std::realloc(import_global_ids, 2 * num_import * sizeof(ZOLTAN_ID_TYPE));
467  write_zoltan_id(item.first, &import_global_ids[2*(num_import-1)]);
468  import_procs = (int*)
469  std::realloc(import_procs, num_import * sizeof(int));
470  import_procs[num_import-1] = item.second.first;
471 
472  }
473 
474  }
475  }
476  int* import_to_part = (int*) std::malloc(num_import * sizeof(int));
477  for (int i = 0; i < num_import; ++i) {
478  import_to_part[i] = import_procs[i];
479  }
480  int* export_to_part = (int*) std::malloc(this->export_node_num * sizeof(int));
481  for (int i = 0; i < this->export_node_num; ++i) {
482  export_to_part[i] = this->export_node_procs[i];
483  }
484 
485  ZOLTAN_ID_PTR import_local_ids = (ZOLTAN_ID_PTR) std::malloc(0);
486  ZOLTAN_ID_PTR export_local_ids = (ZOLTAN_ID_PTR) std::malloc(0);
487 
488  this->setZoltanNodeMigration();
489 
490  this->distribute(
491  1,
492  num_import,
493  import_global_ids,
494  import_local_ids,
495  import_procs,
496  import_to_part,
497  export_local_ids,
498  export_to_part);
