67  // Note : Zoltan should be able to directly use unsigned long as ids.
68  // However, this require to set up compile time flags to Zoltan, so we
69  // use 2 entries of the default zoltan unsigned int instead, to make it
70  // compatible with any Zoltan installation.
71  // Passing compile flags to the custom embedded CMake Zoltan
72  // installation might also be possible.
73  unsigned long node_id(const ZOLTAN_ID_PTR global_ids) {
74  return ((unsigned long) (global_ids[0] << 16)) | global_ids[1];
