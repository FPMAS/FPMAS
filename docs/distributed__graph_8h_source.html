126  ::link(NodeId source, NodeId target, ArcId arcId, LayerId layerId) {
127  if(this->getNodes().count(source) > 0) {
128  // Source is local
129  Node<std::unique_ptr<SyncData<T,N,S>>, N>* sourceNode
130  = this->getNode(source);
131  if(this->getNodes().count(target) > 0) {
132  // Source and Target are local, completely local operation
133  Node<std::unique_ptr<SyncData<T,N,S>>, N>* targetNode
134  = this->getNode(target);
135  return this->Graph<std::unique_ptr<SyncData<T,N,S>>, N>::link(
136  sourceNode, targetNode, arcId, layerId
137  );
138  } else {
139  // Target is distant
140  GhostNode<T, N, S>* targetNode
141  = this->getGhost().getNode(target);
142  this->syncMode.initLink(source, target, arcId, layerId);
143  // link local -> distant
144  auto arc = this->getGhost().link(
145  sourceNode, targetNode, arcId, layerId
146  );
147  this->syncMode.notifyLinked(arc);
148  return arc;
149  }
150  } else {
151  // Source is distant
152  if(this->getNodes().count(target) > 0) {
153  // Source is local
154  GhostNode<T, N, S>* sourceNode
155  = this->getGhost().getNode(source);
156  Node<std::unique_ptr<SyncData<T,N,S>>, N>* targetNode
157  = this->getNode(target);
158  this->syncMode.initLink(source, target, arcId, layerId);
159  // link distant -> local
160  auto arc = this->getGhost().link(
161  sourceNode, targetNode, arcId, layerId
162  );
163  this->syncMode.notifyLinked(arc);
164  return arc;
165  }
166  else {
167  // Source and target are distant
168  // Allowed only if the two nodes are ghosts on this graph
169  // TODO: better exception handling
