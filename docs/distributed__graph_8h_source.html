95  // Note : Zoltan should be able to directly use unsigned long as ids.
96  // However, this require to set up compile time flags to Zoltan, so we
97  // use 2 entries of the default zoltan unsigned int instead, to make it
98  // compatible with any Zoltan installation.
99  // Passing compile flags to the custom embedded CMake Zoltan
100  // installation might also be possible.
101  unsigned long node_id(const ZOLTAN_ID_PTR global_ids) {
102  return ((unsigned long) (global_ids[0] << 16)) | global_ids[1];
103  }
