343  template<class T, template<typename> class S> void DistributedGraph<T, S>::distribute() {
344  int changes;
345  int num_gid_entries;
346  int num_lid_entries;
347 
348  int num_import;
349  ZOLTAN_ID_PTR import_global_ids;
350  ZOLTAN_ID_PTR import_local_ids;

355  ZOLTAN_ID_PTR export_local_ids;
356 
357  // Prepares Zoltan to migrate nodes
358  // Must be set up from there, because LB_Partition can call
359  // obj_size_multi_fn when repartitionning
360  this->setZoltanNodeMigration();
361 
362  // Computes Zoltan partitioning
363  int err = this->zoltan.LB_Partition(
364  changes,
365  num_gid_entries,
366  num_lid_entries,
367  num_import,
368  import_global_ids,
369  import_local_ids,
370  import_procs,
371  import_to_part,
372  this->export_node_num,
373  this->export_node_global_ids,
374  export_local_ids,
375  this->export_node_procs,
376  export_to_part
377  );
378 
379 
380  if(changes > 0) {
381 
382  // Migrate nodes from the load balancing
383  // Arcs to export are computed in the post_migrate_pp_fn step.
384  this->zoltan.Migrate(
385  num_import,
386  import_global_ids,
387  import_local_ids,
388  import_procs,
389  import_to_part,
390  this->export_node_num,
391  this->export_node_global_ids,
392  export_local_ids,
393  this->export_node_procs,
394  export_to_part
395  );
396 
397  // Prepares Zoltan to migrate arcs associated to the exported
398  // nodes.
399  this->setZoltanArcMigration();
400 
401  // Unused buffer
402  ZOLTAN_ID_TYPE export_arcs_local_ids[0];
403 
404  // Arcs to import
405  this->zoltan.Migrate(
406  -1,
407  NULL,
408  NULL,
409  NULL,
410  NULL,
411  this->export_arcs_num,
412  this->export_arcs_global_ids,
413  export_arcs_local_ids,
414  this->export_arcs_procs,
415  this->export_arcs_procs // parts = procs
416  );
417  std::free(this->export_arcs_global_ids);
418  std::free(this->export_arcs_procs);
419  }
420

422  this->zoltan.LB_Free_Part(

424  &import_local_ids,

433  &export_to_part
434  );
435

437  this->proxy.synchronize();
438

440  // REPARTITION will be used for all other calls.
441  this->zoltan.Set_Param("LB_APPROACH", "REPARTITION");

443  this->synchronize();
