84  // Note : Zoltan should be able to directly use unsigned long as ids.
85  // However, this require to set up compile time flags to Zoltan, so we
86  // use 2 entries of the default zoltan unsigned int instead, to make it
87  // compatible with any Zoltan installation.
88  // Passing compile flags to the custom embedded CMake Zoltan
89  // installation might also be possible.
90  unsigned long node_id(const ZOLTAN_ID_PTR global_ids) {
91  return ((unsigned long) (global_ids[0] << 16)) | global_ids[1];
92  }
