402  template<class T, SYNC_MODE, int N>
403  Arc<std::unique_ptr<SyncData<T>>, N>* DistributedGraph<T, S, N>
404  ::link(NodeId source, NodeId target, ArcId arcId, LayerId layerId) {
405  if(this->getNodes().count(source) > 0) {
406  // Source is local
407  Node<std::unique_ptr<SyncData<T>>, N>* sourceNode
408  = this->getNode(source);
409  if(this->getNodes().count(target) > 0) {
410  // Source and Target are local, completely local operation
411  Node<std::unique_ptr<SyncData<T>>, N>* targetNode
412  = this->getNode(target);
413  return this->Graph<std::unique_ptr<SyncData<T>>, N>::link(
414  sourceNode, targetNode, arcId, layerId
415  );
416  } else {
417  // Target is distant
418  GhostNode<T, N, S>* targetNode
419  = this->getGhost().getNode(target);
420  syncMode.initLink(source, target, arcId, layerId);
421  // link local -> distant
422  auto arc = this->getGhost().link(
423  sourceNode, targetNode, arcId, layerId
424  );
425  syncMode.notifyLinked(arc);
426  return arc;
427  }
428  } else {
429  // Source is distant
430  if(this->getNodes().count(target) > 0) {
431  // Source is local
432  GhostNode<T, N, S>* sourceNode
433  = this->getGhost().getNode(source);
434  Node<std::unique_ptr<SyncData<T>>, N>* targetNode
435  = this->getNode(target);
436  syncMode.initLink(source, target, arcId, layerId);
437  // link distant -> local
438  auto arc = this->getGhost().link(
439  sourceNode, targetNode, arcId, layerId
440  );
441  syncMode.notifyLinked(arc);
442  return arc;
443  }
444  else {
445  // Source and target are distant
446  // Allowed only if the two nodes are ghosts on this graph
447  // TODO: better exception handling
448  GhostNode<T, N, S>* sourceNode
449  = this->getGhost().getNode(source);
450  GhostNode<T, N, S>* targetNode
451  = this->getGhost().getNode(target);
452 
453  syncMode.initLink(source, target, arcId, layerId);
454  // link distant -> distant
455  auto arc = this->getGhost().link(
456  sourceNode, targetNode, arcId, layerId
457  );
458  syncMode.notifyLinked(arc);
459  return arc;
460  }
461  }

513  template<class T, SYNC_MODE, int N> void DistributedGraph<T, S, N>::distribute(
514  int changes,
515  int num_import,
516  ZOLTAN_ID_PTR import_global_ids,
517  ZOLTAN_ID_PTR import_local_ids,
518  int* import_procs,
519  int* import_to_part,
520  ZOLTAN_ID_PTR export_local_ids,
521  int* export_to_part
522  ) {
523  if(changes > 0) {
524 
525  FPMAS_LOGV(mpiCommunicator.getRank(), "Dist", "Zoltan Node migration...");
526  // Migrate nodes from the load balancing
527  // Arcs to export are computed in the post_migrate_pp_fn step.
528  this->zoltan.Migrate(
529  num_import,
530  import_global_ids,
531  import_local_ids,
532  import_procs,

535  this->export_node_global_ids,
536  export_local_ids,
537  this->export_node_procs,
538  export_to_part
539  );
540  FPMAS_LOGV(mpiCommunicator.getRank(), "Dist", "Zoltan Node migration done.");
541 
542  // Prepares Zoltan to migrate arcs associated to the exported
543  // nodes.
544  this->setZoltanArcMigration();
545 
546  // Unused buffer
547  ZOLTAN_ID_TYPE export_arcs_local_ids[0];
548 
549  FPMAS_LOGV(mpiCommunicator.getRank(), "Dist", "Zoltan Arc migration...");
550  // Arcs to import
551  this->zoltan.Migrate(
552  -1,
553  NULL,
554  NULL,
555  NULL,
556  NULL,
557  this->export_arcs_num,
558  this->export_arcs_global_ids,
559  export_arcs_local_ids,
560  this->export_arcs_procs,
561  this->export_arcs_procs // parts = procs
562  );
563  FPMAS_LOGV(mpiCommunicator.getRank(), "Dist", "Zoltan Arc migration done.");
564  std::free(this->export_arcs_global_ids);
565  std::free(this->export_arcs_procs);
566  }
567 
568  // Frees the zoltan buffers
569  this->zoltan.LB_Free_Part(
570  &import_global_ids,
571  &import_local_ids,
572  &import_procs,
573  &import_to_part
574  );
575 
576  this->zoltan.LB_Free_Part(
577  &this->export_node_global_ids,
578  &export_local_ids,
579  &this->export_node_procs,
580  &export_to_part
581  );
582 
583  // Updates nodes locations according to last migrations
584  this->proxy.synchronize();
585 
586  this->synchronize();
587

602  template<class T, SYNC_MODE, int N> void DistributedGraph<T, S, N>::distribute(
603  std::unordered_map<NodeId, std::pair<int, int>> partition
604  ) {
605 
606  // Import lists
607  int num_import = 0;
608  ZOLTAN_ID_PTR import_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
609  int* import_procs = (int*) std::malloc(0);
610 
611  // Export lists
612  this->export_node_num = 0;
613  this->export_node_global_ids = (ZOLTAN_ID_PTR) std::malloc(0);
614  this->export_node_procs = (int*) std::malloc(0);
615 
616  for(auto item : partition) {
617  if(item.second.first == this->getMpiCommunicator().getRank()) {
618  if(item.second.second != this->getMpiCommunicator().getRank()) {
619  this->export_node_num++;
620  this->export_node_global_ids = (ZOLTAN_ID_PTR)
621  std::realloc(this->export_node_global_ids, 2 * this->export_node_num * sizeof(ZOLTAN_ID_TYPE));
622 
623  write_zoltan_id(item.first, &this->export_node_global_ids[2*(this->export_node_num-1)]);
624  this->export_node_procs = (int*)
625  std::realloc(this->export_node_procs, this->export_node_num * sizeof(int));
626  this->export_node_procs[this->export_node_num-1] = item.second.second;
627  }
628 
629  } else {
630  if(item.second.second == this->getMpiCommunicator().getRank()) {
631  num_import++;
632  import_global_ids = (ZOLTAN_ID_PTR)
633  std::realloc(import_global_ids, 2 * num_import * sizeof(ZOLTAN_ID_TYPE));
634  write_zoltan_id(item.first, &import_global_ids[2*(num_import-1)]);
635  import_procs = (int*)
636  std::realloc(import_procs, num_import * sizeof(int));
637  import_procs[num_import-1] = item.second.first;
638 
639  }
640 
641  }
642  }
643  int* import_to_part = (int*) std::malloc(num_import * sizeof(int));
644  for (int i = 0; i < num_import; ++i) {
645  import_to_part[i] = import_procs[i];
646  }
647  int* export_to_part = (int*) std::malloc(this->export_node_num * sizeof(int));
648  for (int i = 0; i < this->export_node_num; ++i) {
649  export_to_part[i] = this->export_node_procs[i];
650  }
651 
652  ZOLTAN_ID_PTR import_local_ids = (ZOLTAN_ID_PTR) std::malloc(0);
653  ZOLTAN_ID_PTR export_local_ids = (ZOLTAN_ID_PTR) std::malloc(0);
654 
655  this->setZoltanNodeMigration();
656 
657  this->distribute(
658  1,
659  num_import,
660  import_global_ids,
661  import_local_ids,
662  import_procs,
663  import_to_part,
664  export_local_ids,
665  export_to_part);
666  }
