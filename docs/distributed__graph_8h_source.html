184  // Note : Zoltan should be able to directly use unsigned long as ids.
185  // However, this require to set up compile time flags to Zoltan, so we
186  // use 2 entries of the default zoltan unsigned int instead, to make it
187  // compatible with any Zoltan installation.
188  // Passing compile flags to the custom embedded CMake Zoltan
189  // installation might also be possible.
190  unsigned long node_id(const ZOLTAN_ID_PTR global_ids) {
191  return ((unsigned long) (global_ids[0] << 16)) | global_ids[1];
192  }

206  global_ids[1] = (node_id & 0xFFFF);

425  for (int i = 0; i < num_ids; ++i) {
426  // Rebuilt node id

428

431  for(int j = 0; j < sizes[i] - 1; j++) {
432  buf[idx[i] + j] = node_str[j];
433  }
434  buf[idx[i] + sizes[i] - 1] = 0; // str final char
435 
436  // Removes entry from the serialization buffer
437  serial_cache.erase(id);
