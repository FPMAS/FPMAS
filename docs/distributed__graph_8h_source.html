141  ::link(node_ptr source, node_ptr target, LayerId layerId) {
142  if(this->getNodes().count(source->getId()) > 0) {
143  // Source is local
144  if(this->getNodes().count(target->getId()) > 0) {
145  // Source and Target are local, completely local operation
146  return this->Graph<std::unique_ptr<SyncData<T,N,S>>, DistributedId, N>::link(
147  source, target, layerId
148  );
149  } else {
150  // Target is distant
151  DistributedId id = this->currentArcId++;
152  this->syncMode.initLink(source->getId(), target->getId(), id, layerId);
153  // link local -> distant
154  auto arc = this->getGhost().link(
155  source, (GhostNode<T, N, S>*) target, id, layerId
156  );
157  this->syncMode.notifyLinked(arc);
158  return arc;
159  }
160  } else {
161  // Source is distant
162  if(this->getNodes().count(target->getId()) > 0) {
163  // Source is local
164  DistributedId id = this->currentArcId++;
165  this->syncMode.initLink(source->getId(), target->getId(), id, layerId);
166  // link distant -> local
167  auto arc = this->getGhost().link(
168  (GhostNode<T, N, S>*) source, target, id, layerId
169  );
170  this->syncMode.notifyLinked(arc);
171  return arc;
172  }
173  else {
174  // Source and target are distant
175  // Allowed only if the two nodes are ghosts on this graph
176  // TODO: better exception handling
177  DistributedId id = this->currentArcId++;
178  this->syncMode.initLink(source->getId(), target->getId(), id, layerId);
179  // link distant -> distant
180  auto arc = this->getGhost().link(
181  (GhostNode<T, N, S>*) source, (GhostNode<T, N, S>*) target, id, layerId
182  );

382  int num_gid_entries;
383 
384  int num_import;
385  ZOLTAN_ID_PTR import_global_ids;
386  ZOLTAN_ID_PTR import_local_ids;
387  int * import_procs;
388  int * import_to_part;
389 
390  int* export_to_part;
391  ZOLTAN_ID_PTR export_local_ids;
392 
393  auto periods = this->scheduler.gatherPeriods(
394  this->getMpiCommunicator().getMpiComm());
395  int free=false;
396  for(auto schedule_period : periods) {
397  for(auto n : this->scheduler.get(schedule_period)) {
398  this->toBalance.insert(n);
399  }
400  if(free) {
401  // Frees the zoltan buffers
402  this->zoltan.LB_Free_Part(
403  &import_global_ids,
404  &import_local_ids,
405  &import_procs,
406  &import_to_part
407  );
408 
409  this->zoltan.LB_Free_Part(
410  &this->export_node_global_ids,
411  &export_local_ids,
412  &this->export_node_procs,
413  &export_to_part
414  );
415  }
416  free = true;
417  balance(
418  num_gid_entries,
419  num_import,
420  import_global_ids,
421  import_local_ids,
422  import_procs,
423  import_to_part,
424  export_local_ids,
425  export_to_part
426  );
427  assert(num_gid_entries == 2);
428  auto nodesToFix = this->scheduler.get(schedule_period);
429 
430  // By default, fixes the local nodes to this proc
431  // Nodes that are not in the "export list" have implicitly been
432  // assigned to this proc, so they should be fixed to it.
433  for(auto nodeToFix : nodesToFix) {
434  this->fixedVertices[nodeToFix->getId()]
435  = this->getMpiCommunicator().getRank();
436 
437  }
438 
439  // Nodes to import are not "owned" by this proc, so they will
440  // be fixed as "exported nodes" by their owning procs
441 
442  // Fixes nodes to export to the part where they should have
443  // been exported. Even if they are not exported yet, the next
444  // iterations of the load balancing algorithm will behave as if
445  // they were already on the corresponding procs.
446  for (int i = 0; i < this->export_node_num; i++) {
447  auto nodeId = zoltan::utils::read_zoltan_id(
448  &this->export_node_global_ids[i * num_gid_entries]);
449  if(this->getNodes().count(nodeId) > 0
450  && nodesToFix.count(this->getNode(nodeId)) > 0) {
451  this->fixedVertices[nodeId] = this->export_node_procs[i];
452  }
453  }
454  }
455  this->fixedVertices.clear();
456  this->toBalance.clear();
457 
458  // The final load balancing is migrated in a single operation
459  FPMAS_LOGV(this->mpiCommunicator.getRank(), "DIST_GRAPH", "Zoltan Node migration...");
460  // Migrate nodes from the load balancing
461  // Arcs to export are computed in the post_migrate_pp_fn step.
462  this->zoltan.Migrate(
463  num_import,
464  import_global_ids,
465  import_local_ids,
466  import_procs,
467  import_to_part,
468  this->export_node_num,
469  this->export_node_global_ids,
470  export_local_ids,
471  this->export_node_procs,
472  export_to_part
473  );
474  FPMAS_LOGV(this->mpiCommunicator.getRank(), "DIST_GRAPH", "Zoltan Node migration done.");
475 
476 
477  this->distributeArcs();
478 
479  // Frees node buffers
480  this->zoltan.LB_Free_Part(
481  &import_global_ids,
482  &import_local_ids,
483  &import_procs,
484  &import_to_part
485  );
486  this->zoltan.LB_Free_Part(
487  &this->export_node_global_ids,
488  &export_local_ids,
489  &this->export_node_procs,
490  &export_to_part
491  );
492

494  // REPARTITION will be used for all other calls.
495  this->zoltan.Set_Param("LB_APPROACH", "REPARTITION");
