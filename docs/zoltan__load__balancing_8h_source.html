189 template<typename T> void num_edges_multi_fn(
190 void *data,
191 int, // num_gid_entries (unused)
192 int, // num_lid_entries (unused)
193 int num_obj,
194 ZOLTAN_ID_PTR, // global_ids (unused)
195 ZOLTAN_ID_PTR, // local_ids (unused)
196 int *num_edges,
197 int * // ierr (unused)
198 ) {
199 ZoltanData<T>* z_data = (ZoltanData<T>*) data;
200 // Initializes `num_obj` empty vectors
201 z_data->edges.resize(num_obj);
202 for(int i = 0; i < num_obj; i++) {
203 auto node = z_data->nodes.at(i);
204 int count = 0;
205 for(auto edge : node->getOutgoingEdges()) {
206 // Considers the neighbor only if the load balancing
207 // algorithm is also applied to it, potentially from an
208 // other process.
209 // The distributed_node_ids list contains the ids of ALL
210 // the nodes that are currently balanced.
211 // It is the responsability of
212 // ZoltanLoadBalancing::balance() to perform communications
213 // required to build this list, before the Zoltan algorithm
214 // is effectively applied.
215 if(z_data->distributed_node_ids.count(edge->getTargetNode()->getId()) > 0) {
216 count++;
217 z_data->edges.at(i).push_back(edge);
218 }
219 }
220 num_edges[i] = count;
221 }
222 }
