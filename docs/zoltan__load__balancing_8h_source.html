199 void *data,
200 int, // num_gid_entries (unused)
201 int, // num_lid_entries (unused)
202 int num_obj,

208 struct NodeHandler {
209 static void handle(
210 ZoltanData<T>* z_data, int& count, int i, DistributedId node_id,
211 api::graph::DistributedNode<T>* target_node, float edge_weight) {
212 auto tgt_id = target_node->getId();
213 // Incoming and outgoing edges are considered from all the
214 // nodes, but should be treated once by all processes. (Notice
215 // that the two nodes might be located on distinct processes)
216 //
217 // Considering the relation `rel` below, an edge is considered
218 // only if `rel(node_id, tgt_id)` is true. Moreover, `rel` is
219 // such as `rel(A, B) == true` XOR `rel(B, A) == true`: this
220 // ensures that each edge is treated exactly once from A or B.
221 // The first comparison on id() should ensure that no
222 // preference is given on A or B for each edge. Moreover,
223 // `rel(A, A) == false` so self edges are ignored.
224 if(node_id.id() > tgt_id.id()
225 || (node_id.id() == tgt_id.id() && node_id.rank() > tgt_id.rank())
226 ) {
227 // Considers the neighbor only if the load balancing
228 // algorithm is also applied to it, potentially from an
229 // other process.
230 // The distributed_node_ids list contains the ids of ALL
231 // the nodes that are currently balanced.
232 // It is the responsability of
233 // ZoltanLoadBalancing::balance() to perform communications
234 // required to build this list, before the Zoltan algorithm
235 // is effectively applied.
236 if(z_data->distributed_node_ids.count(tgt_id) > 0) {
237 auto& target_nodes = z_data->target_nodes[i];
238 auto& edge_weights = z_data->edge_weights[i];
239 auto it = std::find(
240 target_nodes.begin(), target_nodes.end(),
241 target_node
242 );
243 if(it == target_nodes.end()) {
244 // No edges between node_id and tgt_id exists:
245 // create a new one
246 count++;
247 target_nodes.push_back(target_node);
248 edge_weights.push_back(edge_weight);
249 } else {
250 // An edge (outgoing or incoming) from node_id to
251 // tgt_id already exists, so the weight of the
252 // current edge is added to the existing one.
253 edge_weights[std::distance(target_nodes.begin(), it)]
254 +=edge_weight;
255 }
256 }
257 }
258 }
259 };
260
261 ZoltanData<T>* z_data = (ZoltanData<T>*) data;
262 // Initializes `num_obj` empty vectors
263 z_data->target_nodes.resize(num_obj);
264 z_data->edge_weights.resize(num_obj);
265 for(int i = 0; i < num_obj; i++) {
266 auto node = z_data->nodes[i];
267 auto node_id = node->getId();
268 int count = 0;
269 for(auto edge : node->getOutgoingEdges())
270 NodeHandler::handle(
271 z_data, count, i, node_id,
272 edge->getTargetNode(), edge->getWeight()
273 );
274 for(auto edge : node->getIncomingEdges())
275 NodeHandler::handle(
276 z_data, count, i, node_id,
277 edge->getSourceNode(), edge->getWeight()
278 );
279
280 num_edges[i] = count;
281 }
282 }
