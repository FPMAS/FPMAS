216  template<typename T, int N, SYNC_MODE> GhostNode<T, N, S>* GhostGraph<T, N, S>
217  ::buildNode(Node<std::unique_ptr<SyncData<T,N,S>>, N>& node, std::set<NodeId> ignoreIds) {
218  FPMAS_LOGD(
219  this->mpiCommunicator.getRank(),
220  "GHOST_GRAPH",
221  "Building ghost node %lu",
222  node.getId()
223  );
224  // Builds the gNode from the original node data
225  GhostNode<T, N, S>* gNode = new GhostNode<T, N, S>(
226  this->localGraph->getMpiCommunicator(),
227  this->localGraph->getProxy(),
228  node
229  );
230  // Register the new GhostNode
231  this->ghostNodes[gNode->getId()] = gNode;
232 
233  for(auto& layer : node.getLayers()) {
234  // Replaces the incomingArcs list by proper GhostArcs
235  for(auto arc : layer.getIncomingArcs()) {
236  auto localSourceNode = arc->getSourceNode();
237  // Builds the ghost arc if :
238  if(
239  // The source node is not ignored (i.e. it is exported in the
240  // current epoch)
241  ignoreIds.count(localSourceNode->getId()) == 0
242  // AND it is not an already built ghost node
243  && this->getNodes().count(localSourceNode->getId()) == 0
244  ) {
245  FPMAS_LOGV(
246  this->mpiCommunicator.getRank(),
247  "GHOST_GRAPH",
248  "Ghost linking %lu : (%lu, %lu)",
249  arc->getId(), localSourceNode->getId(), gNode->getId()
250  );
251  this->link(localSourceNode, gNode, arc->getId(), arc->layer);
252  }
253  }
254 
255  // Replaces the outgoingArcs list by proper GhostArcs
256  for(auto arc : layer.getOutgoingArcs()) {
257  auto localTargetNode = arc->getTargetNode();
258  // Same as above
259  if(ignoreIds.count(localTargetNode->getId()) == 0
260  && this->getNodes().count(localTargetNode->getId()) == 0) {
261  FPMAS_LOGV(
262  this->mpiCommunicator.getRank(),
263  "GHOST_GRAPH",
264  "Ghost linking %lu : (%lu, %lu)",

266  );
267  this->link(gNode, localTargetNode, arc->getId(), arc->layer);
268  }
269  }
