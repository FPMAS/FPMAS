212  ::buildNode(Node<std::unique_ptr<SyncData<T,N,S>>, N>& node, std::set<NodeId> ignoreIds) {
213  // Builds the gNode from the original node data
214  GhostNode<T, N, S>* gNode = new GhostNode<T, N, S>(
215  this->localGraph->getMpiCommunicator(),
216  this->localGraph->getProxy(),
217  node
218  );
219  // Register the new GhostNode
220  this->ghostNodes[gNode->getId()] = gNode;
221 
222  // Replaces the incomingArcs list by proper GhostArcs
223  for(auto layer : node.getLayers()) {
224  for(auto arc : layer.getIncomingArcs()) {
225  auto localSourceNode = arc->getSourceNode();
226  // Builds the ghost arc if :
227  if(
228  // The source node is not ignored (i.e. it is exported in the
229  // current epoch)
230  ignoreIds.count(localSourceNode->getId()) == 0
231  // AND it is not an already built ghost node
232  && this->getNodes().count(localSourceNode->getId()) == 0
233  )
234  this->link(localSourceNode, gNode, arc->getId(), arc->layer);
235  }
236  }
237 
238  // Replaces the outgoingArcs list by proper GhostArcs
239  for(auto layer : node.getLayers()) {
240  for(auto arc : layer.getOutgoingArcs()) {
241  auto localTargetNode = arc->getTargetNode();
242  // Same as above
243  if(ignoreIds.count(localTargetNode->getId()) == 0
244  && this->getNodes().count(localTargetNode->getId()) == 0)
245  this->link(gNode, localTargetNode, arc->getId(), arc->layer);
246  }
247  }
