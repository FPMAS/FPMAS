180  template<class T, template<typename> class S> GhostNode<T, S>* GhostGraph<T, S>::buildNode(Node<SyncData<T>> node, std::set<unsigned long> ignoreIds) {
181  // Copy the gNode from the original node, including arcs data
182  GhostNode<T, S>* gNode = new GhostNode<T, S>(node);
183  // Register the new GhostNode
184  this->ghostNodes[gNode->getId()] = gNode;
185 
186  // Replaces the incomingArcs list by proper GhostArcs
187  gNode->incomingArcs.clear();
188  for(auto arc : node.getIncomingArcs()) {
189  Node<SyncData<T>>* localSourceNode = arc->getSourceNode();
190  // Builds the ghost arc if :
191  if(
192  // The source node is not ignored (i.e. it is exported in the
193  // current epoch)
194  ignoreIds.count(localSourceNode->getId()) == 0
195  // AND it is not an already built ghost node
196  && this->getNodes().count(localSourceNode->getId()) == 0
197  )
198  this->link(localSourceNode, gNode, arc->getId());
199  }
200 
201  // Replaces the outgoingArcs list by proper GhostArcs
202  gNode->outgoingArcs.clear();
203  for(auto arc : node.getOutgoingArcs()) {
204  Node<SyncData<T>>* localTargetNode = arc->getTargetNode();
205  // Same as above
206  if(ignoreIds.count(localTargetNode->getId()) == 0
207  && this->getNodes().count(localTargetNode->getId()) == 0)
208  this->link(gNode, localTargetNode, arc->getId());
209  }
210
