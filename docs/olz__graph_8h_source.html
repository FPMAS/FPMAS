239  ::buildNode(

241  std::set<DistributedId> ignoreIds
242  ) {
243  FPMAS_LOGD(
244  this->mpiCommunicator.getRank(),
245  "GHOST_GRAPH",
246  "Building ghost node %lu",
247  node.getId()
248  );
249  // Builds the gNode from the original node data
250  GhostNode<T, N, S>* gNode = new GhostNode<T, N, S>(
251  this->localGraph->getMpiCommunicator(),
252  this->localGraph->getProxy(),
253  node
254  );
255  // Register the new GhostNode
256  this->ghostNodes[gNode->getId()] = gNode;
257 
258  for(auto& layer : node.getLayers()) {
259  // Replaces the incomingArcs list by proper GhostArcs
260  for(auto arc : layer.getIncomingArcs()) {
261  auto localSourceNode = arc->getSourceNode();
262  // Builds the ghost arc if :
263  if(
264  // The source node is not ignored (i.e. it is exported in the
265  // current epoch)
266  ignoreIds.count(localSourceNode->getId()) == 0
267  // AND it is not an already built ghost node
268  && this->getNodes().count(localSourceNode->getId()) == 0
269  ) {
270  FPMAS_LOGV(
271  this->mpiCommunicator.getRank(),
272  "GHOST_GRAPH",
273  "Ghost linking %lu : (%lu, %lu)",
274  arc->getId(), localSourceNode->getId(), gNode->getId()
275  );
276  this->link(localSourceNode, gNode, arc->getId(), arc->layer);
277  }
278  }
279 
280  // Replaces the outgoingArcs list by proper GhostArcs
281  for(auto arc : layer.getOutgoingArcs()) {
282  auto localTargetNode = arc->getTargetNode();
283  // Same as above
284  if(ignoreIds.count(localTargetNode->getId()) == 0
285  && this->getNodes().count(localTargetNode->getId()) == 0) {
286  FPMAS_LOGV(
