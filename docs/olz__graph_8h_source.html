186  template<class T, template<typename> class S> GhostNode<T, S>* GhostGraph<T, S>::buildNode(Node<SyncData<T>> node, std::set<unsigned long> ignoreIds) {
187  // Copy the gNode from the original node, including arcs data
188  GhostNode<T, S>* gNode = new GhostNode<T, S>(node);
189  // Register the new GhostNode
190  this->ghostNodes[gNode->getId()] = gNode;
191 
192  // Replaces the incomingArcs list by proper GhostArcs
193  gNode->incomingArcs.clear();
194  for(auto arc : node.getIncomingArcs()) {
195  Node<SyncData<T>>* localSourceNode = arc->getSourceNode();
196  // Builds the ghost arc if :
197  if(
198  // The source node is not ignored (i.e. it is exported in the
199  // current epoch)
200  ignoreIds.count(localSourceNode->getId()) == 0
201  // AND it is not an already built ghost node
202  && this->getNodes().count(localSourceNode->getId()) == 0
203  )
204  this->link(localSourceNode, gNode, arc->getId());
205  }
206 
207  // Replaces the outgoingArcs list by proper GhostArcs
208  gNode->outgoingArcs.clear();
209  for(auto arc : node.getOutgoingArcs()) {
210  Node<SyncData<T>>* localTargetNode = arc->getTargetNode();
211  // Same as above
212  if(ignoreIds.count(localTargetNode->getId()) == 0
213  && this->getNodes().count(localTargetNode->getId()) == 0)
214  this->link(gNode, localTargetNode, arc->getId());
215  }
