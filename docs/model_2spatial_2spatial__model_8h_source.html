424  void SpatialAgentBase<AgentType, CellType, Derived>::updateLocation(CellType* cell) {
425  // Links to new location
426  this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
427 
428  // Unlinks obsolete location
429  auto location = this->template outNeighbors<api::model::Cell>(
430  SpatialModelLayers::LOCATION);
431  if(location.count() > 0)
432  this->model()->unlink(location.at(0).edge());
433 
434  // Unlinks obsolete mobility field
435  for(auto _cell : this->template outNeighbors<api::model::Cell>(
436  SpatialModelLayers::MOVE)) {
437  this->model()->unlink(_cell.edge());
438  }
439 
440  // Unlinks obsolete perception field
441  for(auto agent : this->template outNeighbors<api::model::Cell>(
442  SpatialModelLayers::PERCEIVE))
443  this->model()->unlink(agent.edge());
444 
445 
446  // Adds the NEW_LOCATION to the mobility/perceptions fields
447  // depending on the current ranges
448  if(this->mobility_range->contains(cell, cell))
449  this->model()->link(this, cell, SpatialModelLayers::MOVE);
450  if(this->perception_range->contains(cell, cell))
451  this->model()->link(this, cell, SpatialModelLayers::PERCEIVE);
452 
453  this->location_id = cell->node()->getId();
