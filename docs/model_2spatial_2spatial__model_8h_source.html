249  template<typename CellType, typename TypeIdBase>
250  void Cell<CellType, TypeIdBase>::init() {
251  bool init_successors;
252 
253  // This has no performance impact
254  auto current_successors
255  = this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR);
256  // Checks if the currently buffered successors are stricly equal to the
257  // current_successors. In this case, there is no need to update the
258  // successors() list
259  if(current_successors.size() == 0 ||
260  current_successors.size() != raw_successors_buffer.size()) {
261  init_successors = false;
262  } else {
263  auto it = current_successors.begin();
264  auto raw_it = raw_successors_buffer.begin();
265  while(it != current_successors.end() && (*it) == *raw_it) {
266  it++;
267  raw_it++;
268  }
269  if(it == current_successors.end()) {
270  init_successors = true;
271  } else {
272  init_successors = false;
273  }
274  }
275 
276  if(!init_successors) {
277  raw_successors_buffer = current_successors;
278  successors_buffer.resize(raw_successors_buffer.size());
279  for(std::size_t i = 0; i < raw_successors_buffer.size(); i++)
280  successors_buffer[i] = dynamic_cast<CellType*>(
281  raw_successors_buffer[i]->getTargetNode()->data().get()
282  );
283  }

628  // Links to new location
629  this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
630 
631  // Unlinks obsolete location
632  auto location = this->template outNeighbors<api::model::Cell>(
633  SpatialModelLayers::LOCATION);
634  if(location.count() > 0)
