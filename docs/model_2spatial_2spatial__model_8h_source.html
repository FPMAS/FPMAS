316 template<typename CellInterface, typename CellType, typename TypeIdBase>
317 void CellBase<CellInterface, CellType, TypeIdBase>::init() {
318 bool init_successors;
319
320 // This has no performance impact
321 auto current_successors
322 = this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR);
323 // Checks if the currently buffered successors are stricly equal to the
324 // current_successors. In this case, there is no need to update the
325 // successors() list
326 if(current_successors.size() == 0 ||
327 current_successors.size() != raw_successors_buffer.size()) {
328 init_successors = false;
329 } else {
330 auto it = current_successors.begin();
331 auto raw_it = raw_successors_buffer.begin();
332 while(it != current_successors.end() && (*it) == *raw_it) {
333 it++;
334 raw_it++;
335 }
336 if(it == current_successors.end()) {
337 init_successors = true;
338 } else {
339 init_successors = false;
340 }
341 }
342
343 if(!init_successors) {
344 raw_successors_buffer = current_successors;
345 successors_buffer.resize(raw_successors_buffer.size());
346 for(std::size_t i = 0; i < raw_successors_buffer.size(); i++)
347 successors_buffer[i] = static_cast<CellType*>(
348 raw_successors_buffer[i]->getTargetNode()->data().get()
349 );
350 }
351
352 std::set<DistributedId> new_location_layer;
353 for(auto agent_edge
354 : this->node()->getIncomingEdges(SpatialModelLayers::NEW_LOCATION))
355 new_location_layer.insert(agent_edge->getSourceNode()->getId());
356
357 // The location of agents that are still linked on the MOVE /
358 // PERCEIVE layer has not been updated, since those layers are
359 // unlinked by SpatialAgent::updateLocation(), unless the current
360 // cell is actually their NEW_LOCATION.
361 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
362 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
363 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
364 }
365 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
366 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
367 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
368 }
369 // All agents already linked on the LOCATION layer didn't update
370 // their location, since LOCATION links are unlinked by
371 // SpatialAgent::updateLocation().
372 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::LOCATION)) {
373 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
374 }
375 }

377 template<typename CellInterface, typename CellType, typename TypeIdBase>
378 void CellBase<CellInterface, CellType, TypeIdBase>::growMobilityField(api::model::Agent* agent) {
379 if(agent->node()->state() == api::graph::LOCAL) {
380 // There is no need to create a temporary NEW_MOVE
381 // link, since the mobilityRange() of the agent is
382 // directly available on the current process.
383 auto* spatial_agent
384 = static_cast<api::model::SpatialAgent<CellType>*>(agent);
385
386 // Valid, since the agent is LOCAL
387 auto location = spatial_agent->locationCell();
388
389 fpmas::model::ReadGuard read_location(location);
390
391 // Cells currently in the move_layer
392 detail::CurrentOutLayer move_layer(agent, SpatialModelLayers::MOVE);
393 for(auto cell : this->bufferedSuccessors()) {
394 if(!move_layer.contains(cell)) {
395 // The cell as not already been linked in the
396 // MOVE layer
397 fpmas::model::ReadGuard read_cell(cell);
398 if(spatial_agent->mobilityRange().contains(
399 location,
400 cell
401 )) {
402 // The cell is contained in the agent
403 // mobility range: add it to its mobility
404 // field
405 move_layer.link(cell);
406 }
407 }
408 }
409 } else {
410 // In this case, the mobility range of the agent is not
411 // available on the current process, so a temporary
412 // NEW_MOVE edge must be created and handled later by
413 // the agent in the SpatialAgent::handleNewMove()
414 // method
415 for(auto cell : this->bufferedSuccessors())
416 this->model()->link(
417 agent,
418 cell,
419 SpatialModelLayers::NEW_MOVE
420 );
421 }

424 template<typename CellInterface, typename CellType, typename TypeIdBase>
425 std::vector<api::model::Cell*> CellBase<CellInterface, CellType, TypeIdBase>::successors() {
426 std::vector<api::model::Cell*> neighbors;
427 for(auto edge : this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR)) {
428 // Assumes that agents on the CELL_SUCCESSOR layer are necessarily
429 // api::model::Cell, so there is no runtime check on the actuel
430 // Agent type (prevents a dynamic_cast)
431 neighbors.push_back(static_cast<api::model::Cell*>(edge->getTargetNode()->data().get()));
432 }

461 FPMAS_LOGD(this->model()->graph().getMpiCommunicator().getRank(), "[CELL]",
462 "%s Updating ranges...",
463 FPMAS_C_STR(this->node()->getId()));
464
465 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::NEW_LOCATION)) {
466 AgentPtr& agent = agent_edge->getSourceNode()->data();
467 updateLocation(agent, agent_edge);
468 growMobilityField(agent);
469 growPerceptionField(agent);

471 // If this node is to be linked in the MOVE or PERCEPTION
472 // field, this has already been done by
473 // LocatedAgent::moveToCell(), so this current location does

476 perception_flags.insert(agent->node()->getId());

481 void CellBase<CellInterface, CellType, TypeIdBase>::handleMove() {
482 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
483 auto agent = agent_edge->getSourceNode();
484 if(
485 // The agent updated its location since the last
486 // DistributedMoveAlgorithm execution
487 no_move_flags.count(agent->getId()) == 0 &&
488 // The current cell was not explored yet
489 move_flags.count(agent->getId()) == 0) {
490 growMobilityField(agent->data());
491 move_flags.insert(agent->getId());
492 }
493 }

512 template<typename CellInterface, typename CellType, typename TypeIdBase>
513 void CellBase<CellInterface, CellType, TypeIdBase>::updatePerceptions(api::model::AgentGroup&) {
514 FPMAS_LOGD(this->model()->graph().getMpiCommunicator().getRank(), "[CELL]",
515 "%s Updating perceptions...",
516 FPMAS_C_STR(this->node()->getId()));
517
518 move_flags.clear();
519 perception_flags.clear();
520 std::vector<api::model::AgentEdge*> perceived_agent_edges =
521 this->node()->getIncomingEdges(SpatialModelLayers::LOCATION);
522 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
523 auto agent = agent_edge->getSourceNode();
524 for(auto perceived_agent_edge : perceived_agent_edges) {
525 auto perceived_agent = perceived_agent_edge->getSourceNode();
526 // Perceptions need to be updated only if either the perceivee
527 // or the perceiver location has been updated since the last
528 // DistributedMoveAlgorithm execution
529 if(
530 this->no_move_flags.count(agent->getId()) == 0 ||
531 this->no_move_flags.count(perceived_agent->getId()) == 0
532 )
533 if(perceived_agent->getId() != agent->getId())
534 this->model()->graph().link(
535 agent, perceived_agent,
536 SpatialModelLayers::PERCEPTION
537 );
538 }
539 }
540 no_move_flags.clear();

544 void CellBase<CellInterface, CellType, TypeIdBase>::growPerceptionField(api::model::Agent* agent) {
545 // Exactly the same process as growMobilityField().
546 if(agent->node()->state() == api::graph::LOCAL) {
547 auto* spatial_agent
548 = static_cast<api::model::SpatialAgent<CellType>*>(agent);
549
550 auto location = spatial_agent->locationCell();
551 fpmas::model::ReadGuard read_location(location);
552
553 detail::CurrentOutLayer perceive_layer(agent, SpatialModelLayers::PERCEIVE);
554 for(auto cell : this->bufferedSuccessors()) {
555 if(!perceive_layer.contains(cell)) {
556 fpmas::model::ReadGuard read_cell(cell);
557 if(spatial_agent->perceptionRange().contains(
558 location,
559 cell
560 )) {
561 perceive_layer.link(cell);
562 }
563 }
564 }
565 } else {
566 for(auto cell : this->bufferedSuccessors())
567 this->model()->link(
568 agent,
569 cell,
570 SpatialModelLayers::NEW_PERCEIVE
571 );
572 }

796 template<typename SpatialAgentInterface, typename AgentType, typename CellType, typename Derived>
797 void SpatialAgentBase<SpatialAgentInterface, AgentType, CellType, Derived>::updateLocation(CellType* cell) {
798 // Links to new location
799 this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
800
801 // Unlinks obsolete location
802 auto location = this->node()->getOutgoingEdges(SpatialModelLayers::LOCATION);
803 if(location.size() > 0)
804 this->model()->unlink(location[0]);
805
806 // Unlinks obsolete mobility field
807 for(auto cell_edge :
808 this->node()->getOutgoingEdges(SpatialModelLayers::MOVE)) {
809 this->model()->unlink(cell_edge);
810 }
811
812 // Unlinks obsolete perception field
813 for(auto cell_edge
814 : this->node()->getOutgoingEdges(SpatialModelLayers::PERCEIVE))
815 this->model()->unlink(cell_edge);
816
817 // Unlinks obsolete perceptions
818 for(auto perception
819 : this->node()->getOutgoingEdges(fpmas::api::model::PERCEPTION))
820 this->model()->unlink(perception);
821
822
823 {
824 // The cell must be read before checking mobility ranges, to
825 // ensure cell data are available (e.g. cell->location())
826 fpmas::model::ReadGuard read(cell);
827
828 // Adds the NEW_LOCATION to the mobility/perceptions fields
829 // depending on the current ranges
830 if(this->mobilityRange().contains(cell, cell))
831 this->model()->link(this, cell, SpatialModelLayers::MOVE);
832 if(this->perceptionRange().contains(cell, cell))
833 this->model()->link(this, cell, SpatialModelLayers::PERCEIVE);
834 }
835
836 this->location_id = cell->node()->getId();
837 this->location_cell_buffer = cell;
838 }

860 detail::CurrentOutLayer move_layer(this, SpatialModelLayers::MOVE);
861

869 move_layer.link(agent);
870 this->model()->unlink(cell_edge);

891 void SpatialAgentBase<SpatialAgentInterface, AgentType, CellType, Derived>::moveTo(DistributedId cell_id) {
892 bool found = false;
893 auto mobility_field = this->template outNeighbors<CellType>(SpatialModelLayers::MOVE);
894 auto it = mobility_field.begin();
895 while(!found && it != mobility_field.end()) {
896 if((*it)->node()->getId() == cell_id) {
897 found=true;
898 } else {
899 it++;
900 }
901 }
902 if(found)
903 this->moveTo(*it);
904 else
905 throw api::model::OutOfMobilityFieldException(this->node()->getId(), cell_id);
906 }
