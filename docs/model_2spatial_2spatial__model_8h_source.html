317 template<typename CellInterface, typename CellType, typename TypeIdBase>
318 void CellBase<CellInterface, CellType, TypeIdBase>::init() {
319 bool init_successors;
320
321 // This has no performance impact
322 auto current_successors
323 = this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR);
324 // Checks if the currently buffered successors are stricly equal to the
325 // current_successors. In this case, there is no need to update the
326 // successors() list
327 if(current_successors.size() == 0 ||
328 current_successors.size() != raw_successors_buffer.size()) {
329 init_successors = false;
330 } else {
331 auto it = current_successors.begin();
332 auto raw_it = raw_successors_buffer.begin();
333 while(it != current_successors.end() && (*it) == *raw_it) {
334 it++;
335 raw_it++;
336 }
337 if(it == current_successors.end()) {
338 init_successors = true;
339 } else {
340 init_successors = false;
341 }
342 }
343
344 if(!init_successors) {
345 raw_successors_buffer = current_successors;
346 successors_buffer.resize(raw_successors_buffer.size());
347 for(std::size_t i = 0; i < raw_successors_buffer.size(); i++)
348 successors_buffer[i] = static_cast<CellType*>(
349 raw_successors_buffer[i]->getTargetNode()->data().get()
350 );
351 }
352
353 std::set<DistributedId> new_location_layer;
354 for(auto agent_edge
355 : this->node()->getIncomingEdges(SpatialModelLayers::NEW_LOCATION))
356 new_location_layer.insert(agent_edge->getSourceNode()->getId());
357
358 // The location of agents that are still linked on the MOVE /
359 // PERCEIVE layer has not been updated, since those layers are
360 // unlinked by SpatialAgent::updateLocation(), unless the current
361 // cell is actually their NEW_LOCATION.
362 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
363 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
364 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
365 }
366 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
367 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
368 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
369 }
370 // All agents already linked on the LOCATION layer didn't update
371 // their location, since LOCATION links are unlinked by
372 // SpatialAgent::updateLocation().
373 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::LOCATION)) {
374 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
375 }
376 }

378 template<typename CellInterface, typename CellType, typename TypeIdBase>
379 void CellBase<CellInterface, CellType, TypeIdBase>::growMobilityField(api::model::Agent* agent) {
380 if(agent->node()->state() == api::graph::LOCAL) {
381 // There is no need to create a temporary NEW_MOVE
382 // link, since the mobilityRange() of the agent is
383 // directly available on the current process.
384 auto* spatial_agent
385 = static_cast<api::model::SpatialAgent<CellType>*>(agent);
386
387 // Valid, since the agent is LOCAL
388 auto location = spatial_agent->locationCell();
389
390 fpmas::model::ReadGuard read_location(location);
391
392 // Cells currently in the move_layer
393 detail::CurrentOutLayer move_layer(agent, SpatialModelLayers::MOVE);
394 for(auto cell : this->bufferedSuccessors()) {
395 if(!move_layer.contains(cell)) {
396 // The cell as not already been linked in the
397 // MOVE layer
398 fpmas::model::ReadGuard read_cell(cell);
399
400 if(spatial_agent->mobilityRange().contains(
401 location,
402 cell
403 )) {
404 // The cell is contained in the agent
405 // mobility range: add it to its mobility
406 // field
407 move_layer.link(cell);
408 }
409 }
410 }
411 } else {
412 // In this case, the mobility range of the agent is not
413 // available on the current process, so a temporary
414 // NEW_MOVE edge must be created and handled later by
415 // the agent in the SpatialAgent::handleNewMove()
416 // method
417 for(auto cell : this->bufferedSuccessors())
418 this->model()->link(
419 agent,
420 cell,
421 SpatialModelLayers::NEW_MOVE
422 );
423 }

427 std::vector<api::model::Cell*> CellBase<CellInterface, CellType, TypeIdBase>::successors() {
428 std::vector<api::model::Cell*> neighbors;
429 for(auto edge : this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR)) {
430 // Assumes that agents on the CELL_SUCCESSOR layer are necessarily
431 // api::model::Cell, so there is no runtime check on the actuel
432 // Agent type (prevents a dynamic_cast)
433 neighbors.push_back(static_cast<api::model::Cell*>(edge->getTargetNode()->data().get()));
434 }

483 void CellBase<CellInterface, CellType, TypeIdBase>::handleMove() {
484 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
485 auto agent = agent_edge->getSourceNode();
486 if(
487 // The agent updated its location since the last
488 // DistributedMoveAlgorithm execution
489 no_move_flags.count(agent->getId()) == 0 &&
490 // The current cell was not explored yet
491 move_flags.count(agent->getId()) == 0) {
492 growMobilityField(agent->data());
493 move_flags.insert(agent->getId());
494 }

514 template<typename CellInterface, typename CellType, typename TypeIdBase>
515 void CellBase<CellInterface, CellType, TypeIdBase>::updatePerceptions(api::model::AgentGroup&) {
516 FPMAS_LOGD(this->model()->graph().getMpiCommunicator().getRank(), "[CELL]",
517 "%s Updating perceptions...",
518 FPMAS_C_STR(this->node()->getId()));
519
520 move_flags.clear();
521 perception_flags.clear();
522 std::vector<api::model::AgentEdge*> perceived_agent_edges =
523 this->node()->getIncomingEdges(SpatialModelLayers::LOCATION);
524 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
525 auto agent = agent_edge->getSourceNode();
526 for(auto perceived_agent_edge : perceived_agent_edges) {
527 auto perceived_agent = perceived_agent_edge->getSourceNode();
528 // Perceptions need to be updated only if either the perceivee
529 // or the perceiver location has been updated since the last
530 // DistributedMoveAlgorithm execution
531 if(
532 this->no_move_flags.count(agent->getId()) == 0 ||
533 this->no_move_flags.count(perceived_agent->getId()) == 0
534 )
535 if(perceived_agent->getId() != agent->getId())
536 this->model()->graph().link(
537 agent, perceived_agent,
538 SpatialModelLayers::PERCEPTION
539 );
540 }
541 }
542 no_move_flags.clear();

547 // Exactly the same process as growMobilityField().

560 location,
561 cell

563 perceive_layer.link(cell);
564 }
565 }

567 } else {

569 this->model()->link(

798 template<typename SpatialAgentInterface, typename AgentType, typename CellType, typename Derived>
799 void SpatialAgentBase<SpatialAgentInterface, AgentType, CellType, Derived>::updateLocation(CellType* cell) {
800 // Links to new location
801 this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
802
803 // Unlinks obsolete location
804 auto location = this->node()->getOutgoingEdges(SpatialModelLayers::LOCATION);
805 if(location.size() > 0)
806 this->model()->unlink(location[0]);
807
808 // Unlinks obsolete mobility field
809 for(auto cell_edge :
810 this->node()->getOutgoingEdges(SpatialModelLayers::MOVE)) {
811 this->model()->unlink(cell_edge);
812 }
813
814 // Unlinks obsolete perception field
815 for(auto cell_edge
816 : this->node()->getOutgoingEdges(SpatialModelLayers::PERCEIVE))
817 this->model()->unlink(cell_edge);
818
819 // Unlinks obsolete perceptions
820 for(auto perception
821 : this->node()->getOutgoingEdges(fpmas::api::model::PERCEPTION))
822 this->model()->unlink(perception);
823
824
825 {
826 // The cell must be read before checking mobility ranges, to
827 // ensure cell data are available (e.g. cell->location())
828 fpmas::model::ReadGuard read(cell);
829
830 // Adds the NEW_LOCATION to the mobility/perceptions fields
831 // depending on the current ranges
832 if(this->mobilityRange().contains(cell, cell))
833 this->model()->link(this, cell, SpatialModelLayers::MOVE);
834 if(this->perceptionRange().contains(cell, cell))
835 this->model()->link(this, cell, SpatialModelLayers::PERCEIVE);
836 }
837
838 this->location_id = cell->node()->getId();
839 this->location_cell_buffer = cell;
840 }

893 void SpatialAgentBase<SpatialAgentInterface, AgentType, CellType, Derived>::moveTo(DistributedId cell_id) {
894 bool found = false;
895 auto mobility_field = this->template outNeighbors<CellType>(SpatialModelLayers::MOVE);
896 auto it = mobility_field.begin();
897 while(!found && it != mobility_field.end()) {
898 if((*it)->node()->getId() == cell_id) {
899 found=true;
900 } else {
901 it++;
902 }
903 }
904 if(found)
905 this->moveTo(*it);
906 else
907 throw api::model::OutOfMobilityFieldException(this->node()->getId(), cell_id);
908 }
