367  void SpatialAgentBase<AgentType, CellType, Derived>::updateLocation(CellType* cell) {
368  // Links to new location
369  this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
370 
371  // Unlinks obsolete location
372  auto location = this->template outNeighbors<api::model::Cell>(
373  SpatialModelLayers::LOCATION);
374  if(location.count() > 0)
375  this->model()->unlink(location.at(0).edge());
376 
377  // Unlinks obsolete mobility field
378  for(auto cell : this->template outNeighbors<api::model::Cell>(
379  SpatialModelLayers::MOVE)) {
380  this->model()->unlink(cell.edge());
381  }
382 
383  // Unlinks obsolete perception field
384  for(auto agent : this->template outNeighbors<api::model::Agent>(
385  SpatialModelLayers::PERCEPTION))
386  this->model()->unlink(agent.edge());
387 
388  // Adds the NEW_LOCATION to the mobility/perceptions fields
389  // depending on the current ranges
390  if(this->mobility_range->contains(cell, cell))
391  this->model()->link(this, cell, SpatialModelLayers::MOVE);
392  if(this->perception_range->contains(cell, cell))
393  this->model()->link(this, cell, SpatialModelLayers::PERCEIVE);
394 
395  this->location_id = cell->node()->getId();
396  }
