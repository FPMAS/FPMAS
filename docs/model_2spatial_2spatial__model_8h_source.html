277 * and more particularly the dynamic_casts that are involved, have
278 * a significant performance cost.
279 * The purpose of the following structures is to drastically reduce
280 * the call to the successors() method, buffering the result only

310 template<typename CellType, typename TypeIdBase>
311 void Cell<CellType, TypeIdBase>::init() {
312 bool init_successors;
313
314 // This has no performance impact
315 auto current_successors
316 = this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR);
317 // Checks if the currently buffered successors are stricly equal to the
318 // current_successors. In this case, there is no need to update the
319 // successors() list
320 if(current_successors.size() == 0 ||
321 current_successors.size() != raw_successors_buffer.size()) {
322 init_successors = false;
323 } else {
324 auto it = current_successors.begin();
325 auto raw_it = raw_successors_buffer.begin();
326 while(it != current_successors.end() && (*it) == *raw_it) {
327 it++;
328 raw_it++;
329 }
330 if(it == current_successors.end()) {
331 init_successors = true;
332 } else {
333 init_successors = false;
334 }
335 }
336
337 if(!init_successors) {
338 raw_successors_buffer = current_successors;
339 successors_buffer.resize(raw_successors_buffer.size());
340 for(std::size_t i = 0; i < raw_successors_buffer.size(); i++)
341 successors_buffer[i] = dynamic_cast<CellType*>(
342 raw_successors_buffer[i]->getTargetNode()->data().get()
343 );
344 }
345
346 std::set<DistributedId> new_location_layer;
347 for(auto agent_edge
348 : this->node()->getIncomingEdges(SpatialModelLayers::NEW_LOCATION))
349 new_location_layer.insert(agent_edge->getSourceNode()->getId());
350
351 // The location of agents that are still linked on the MOVE /
352 // PERCEIVE layer has not been updated, since those layers are
353 // unlinked by SpatialAgent::updateLocation(), unless the current
354 // cell is actually their NEW_LOCATION.
355 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
356 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
357 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
358 }
359 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
360 if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
361 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
362 }
363 // All agents already linked on the LOCATION layer didn't update
364 // their location, since LOCATION links are unlinked by
365 // SpatialAgent::updateLocation().
366 for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::LOCATION)) {
367 this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
368 }

371 template<typename CellType, typename TypeIdBase>
372 void Cell<CellType, TypeIdBase>::growMobilityField(api::model::Agent* agent) {
373 if(agent->node()->state() == api::graph::LOCAL) {
374 // There is no need to create a temporary NEW_MOVE
375 // link, since the mobilityRange() of the agent is
376 // directly available on the current process.
377 auto* spatial_agent
378 = dynamic_cast<api::model::SpatialAgent<CellType>*>(agent);
379
380 // Valid, since the agent is LOCAL
381 auto location = spatial_agent->locationCell();
382
383 fpmas::model::ReadGuard read_location(location);
384
385 // Cells currently in the move_layer
386 detail::CurrentOutLayer move_layer(agent, SpatialModelLayers::MOVE);
387 for(auto cell : this->bufferedSuccessors()) {
388 if(!move_layer.contains(cell)) {
389 // The cell as not already been linked in the
390 // MOVE layer
391 fpmas::model::ReadGuard read_cell(cell);
392 if(spatial_agent->mobilityRange().contains(
393 location,
394 cell
395 )) {
396 // The cell is contained in the agent
397 // mobility range: add it to its mobility
398 // field
399 move_layer.link(cell);
400 }
401 }
402 }
403 } else {
404 // In this case, the mobility range of the agent is not
405 // available on the current process, so a temporary
406 // NEW_MOVE edge must be created and handled later by
407 // the agent in the SpatialAgent::handleNewMove()
408 // method
409 for(auto cell : this->bufferedSuccessors())
410 this->model()->link(
411 agent,
412 cell,
413 SpatialModelLayers::NEW_MOVE
414 );
415 }

642 template<typename AgentType, typename CellType, typename Derived>
643 void SpatialAgentBase<AgentType, CellType, Derived>::updateLocation(CellType* cell) {
644 // Links to new location
645 this->model()->link(this, cell, SpatialModelLayers::NEW_LOCATION);
646
647 // Unlinks obsolete location
648 auto location = this->node()->getOutgoingEdges(SpatialModelLayers::LOCATION);
649 if(location.size() > 0)
650 this->model()->unlink(location[0]);
651
652 // Unlinks obsolete mobility field
653 for(auto cell_edge :
654 this->node()->getOutgoingEdges(SpatialModelLayers::MOVE)) {
655 this->model()->unlink(cell_edge);
656 }
657
658 // Unlinks obsolete perception field
659 for(auto cell_edge
660 : this->node()->getOutgoingEdges(SpatialModelLayers::PERCEIVE))
661 this->model()->unlink(cell_edge);
662
663 // Unlinks obsolete perceptions
664 for(auto perception
665 : this->node()->getOutgoingEdges(fpmas::api::model::PERCEPTION))
666 this->model()->unlink(perception);
667
668
669 // Adds the NEW_LOCATION to the mobility/perceptions fields
670 // depending on the current ranges
671 if(this->mobilityRange().contains(cell, cell))
672 this->model()->link(this, cell, SpatialModelLayers::MOVE);
673 if(this->perceptionRange().contains(cell, cell))
674 this->model()->link(this, cell, SpatialModelLayers::PERCEIVE);
675
676 this->location_id = cell->node()->getId();
677 this->location_cell_buffer = cell;
678 }

704 void SpatialAgentBase<AgentType, CellType, Derived>::handleNewMove() {
705
706 detail::CurrentOutLayer move_layer(this, SpatialModelLayers::MOVE);
707
708 auto location = this->locationCell();
709 fpmas::model::ReadGuard read_location(location);
710 for(auto cell_edge : this->node()->getOutgoingEdges(SpatialModelLayers::NEW_MOVE)) {
711 auto* agent = cell_edge->getTargetNode()->data().get();

713 if(!move_layer.contains(agent)
714 && this->mobilityRange().contains(location, dynamic_cast<CellType*>(agent)))
715 move_layer.link(agent);
716 this->model()->unlink(cell_edge);
717 }

738 bool found = false;
739 auto mobility_field = this->template outNeighbors<CellType>(SpatialModelLayers::MOVE);
740 auto it = mobility_field.begin();
741 while(!found && it != mobility_field.end()) {
742 if((*it)->node()->getId() == cell_id) {
743 found=true;
744 } else {
745 it++;
746 }
747 }
748 if(found)
749 this->moveTo(*it);
