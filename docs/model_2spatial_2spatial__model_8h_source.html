312  template<typename CellType, typename TypeIdBase>
313  void Cell<CellType, TypeIdBase>::init() {
314  bool init_successors;
315 
316  // This has no performance impact
317  auto current_successors
318  = this->node()->getOutgoingEdges(SpatialModelLayers::CELL_SUCCESSOR);
319  // Checks if the currently buffered successors are stricly equal to the
320  // current_successors. In this case, there is no need to update the
321  // successors() list
322  if(current_successors.size() == 0 ||
323  current_successors.size() != raw_successors_buffer.size()) {
324  init_successors = false;
325  } else {
326  auto it = current_successors.begin();
327  auto raw_it = raw_successors_buffer.begin();
328  while(it != current_successors.end() && (*it) == *raw_it) {
329  it++;
330  raw_it++;
331  }
332  if(it == current_successors.end()) {
333  init_successors = true;
334  } else {
335  init_successors = false;
336  }
337  }
338 
339  if(!init_successors) {
340  raw_successors_buffer = current_successors;
341  successors_buffer.resize(raw_successors_buffer.size());
342  for(std::size_t i = 0; i < raw_successors_buffer.size(); i++)
343  successors_buffer[i] = dynamic_cast<CellType*>(
344  raw_successors_buffer[i]->getTargetNode()->data().get()
345  );
346  }
347 
348  std::set<DistributedId> new_location_layer;
349  for(auto agent_edge
350  : this->node()->getIncomingEdges(SpatialModelLayers::NEW_LOCATION))
351  new_location_layer.insert(agent_edge->getSourceNode()->getId());
352 
353  // The location of agents that are still linked on the MOVE /
354  // PERCEIVE layer has not been updated, since those layers are
355  // unlinked by SpatialAgent::updateLocation(), unless the current
356  // cell is actually their NEW_LOCATION.
357  for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::MOVE)) {
358  if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
359  this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
360  }
361  for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::PERCEIVE)) {
362  if(new_location_layer.count(agent_edge->getSourceNode()->getId())==0)
363  this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
364  }
365  // All agents already linked on the LOCATION layer didn't update
366  // their location, since LOCATION links are unlinked by
367  // SpatialAgent::updateLocation().
368  for(auto agent_edge : this->node()->getIncomingEdges(SpatialModelLayers::LOCATION)) {
369  this->no_move_flags.insert(agent_edge->getSourceNode()->getId());
370  }
