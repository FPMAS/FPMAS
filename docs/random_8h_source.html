281 std::vector<std::pair<int, std::size_t>> sample_indexes(
282 api::communication::MpiCommunicator& comm,
283 std::size_t local_items_count, std::size_t n,
284 Generator_t& gen
285 ) {
286 communication::TypedMpi<std::size_t> int_mpi(comm);
287 // Item count on each process
288 std::vector<std::size_t> item_counts
289 = int_mpi.allGather(local_items_count);
290 std::size_t total_item_count = std::accumulate(
291 item_counts.begin(), item_counts.end(), 0);
292 n = std::min(n, total_item_count);
293
294 // (process, local_index)
295 std::vector<std::pair<int, std::size_t>> result_indexes(n);
296
297 /* Initializes the indexes reservoir */
298 int p = 0;
299 std::size_t local_index = 0;
300 for(std::size_t i = 0; i < n; i++) {
301 if(local_index >= item_counts[p]) {
302 p++;
303 local_index = 0;
304 }
305 result_indexes[i] = {p, local_index++};
306 }
307 /**************************************/
308
309 std::size_t i = n;
310 while(p != (int) item_counts.size()) {
311 while(local_index != item_counts[p]) {
312 std::pair<int, std::size_t> index(p, local_index++);
313 UniformIntDistribution<std::size_t> random_k(0, i);
314 std::size_t k = random_k(gen);
315 if(k < n)
316 result_indexes[k] = index;
317 ++i;
318 }
319 p++;
320 local_index=0;
321 }
322
323 return result_indexes;
324 }
