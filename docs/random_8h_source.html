182 std::unordered_map<int, std::size_t> requests;
183

292 std::vector<Index_t> sample_indexes(
293 Index_t begin, Index_t end,
294 std::size_t n,
295 Generator_t& gen
296 ) {
297 // (process, local_index)
298 std::vector<Index_t> result_indexes(n);
299
300 /* Initializes the indexes reservoir */
301 Index_t p = begin;
302 for(std::size_t i = 0; i < n; i++) {
303 result_indexes[i] = p++;
304 }

306

309 UniformIntDistribution<std::size_t> random_k(0, i);

312 result_indexes[k] = p;

315 }

541 Index<K> Index<K>::operator+(std::size_t n) const {
542 Index<K> index = *this;
543 // index.id->second <=> index.item_counts[index.key()]
544 if(index._offset+n < index.it->second) {
545 // Stay in current process, nothing to do
546 index._offset+=n;
547 } else {
548 // Need to go to next process
549 n -= index.it->second - index.offset();
550 index._offset = 0;
551 ++index.it;
552 while(n >= index.it->second) {
553 n-= index.it->second;
554 ++index.it;
555 }
556 index._offset = n;

562 std::size_t Index<K>::distance(
563 const Index<K>& i1, const Index<K>& i2) {
564 if(i1.it != i1.item_counts.end() && i2.it != i2.item_counts.end()) {
565 if(i1.key() == i2.key()) {
566 return i2.offset() - i1.offset();
567 } else {
568 std::size_t result = i1.it->second - i1.offset();
569 auto it = i1.it;
570 ++it;
571 while(it->first != i2.it->first) {
572 result+=it->second;
573 ++it;
574 }
575 result+=i2.offset();
576 return result;
577 }
578 }
579 if(i1.it == i1.item_counts.end() && i2.it == i2.item_counts.end())
580 return 0;
581 // Since i2 reachable from i1, only i2 can be end (if i1 is end and not
582 // i2, unexpected result)

585 auto it = i1.it;
586 ++it;
587 while(it != i1.item_counts.end()) {

589 ++it;
590 }
