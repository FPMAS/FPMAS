148 void GridAgent<AgentType, GridCellType, Derived>::moveTo(DiscretePoint point) {
149 bool found = false;
150 auto mobility_field = this->template outNeighbors<GridCellType>(SpatialModelLayers::MOVE);
151 auto it = mobility_field.begin();
152 while(!found && it != mobility_field.end()) {
153 if((*it)->location() == point) {
154 found=true;
155 } else {
156 it++;
157 }
158 }
159 if(found)
160 this->moveTo(*it);
161 else
162 throw api::model::OutOfMobilityFieldException(this->node()->getId(), this->locationPoint(), point);

333 template<typename CellType>
334 void GridAgentBuilder<CellType>::build(
335 api::model::SpatialModel<CellType>& model,
336 api::model::GroupList groups,
337 std::function<api::model::SpatialAgent<CellType>*()> factory,
338 api::model::SpatialAgentMapping<api::model::GridCell>& agent_mapping
339 ) {
340 // Count of items by DiscretePoint (no entry <=> 0)
341 std::map<DiscretePoint, std::size_t> item_counts;
342 for(auto cell : model.cells()) {
343 std::size_t count = agent_mapping.countAt(cell);
344 if(count > 0)
345 // Does not insert useless null entries
346 item_counts[cell->location()]=count;
347 }
348
349 // Gather item counts from all processes
350 communication::TypedMpi<decltype(item_counts)> item_counts_mpi(
351 model.getMpiCommunicator());
352 item_counts =
353 communication::all_reduce(item_counts_mpi, item_counts, utils::Concat());
354
355 // Contains the cumulative sum of item_counts entries
356 std::map<DiscretePoint, std::size_t> item_counts_sum;
357
358 // Loop ordered by DiscretePoint
359 std::size_t sum = 0;
360 for(auto item : item_counts) {
361 sum+=item.second;
362 item_counts_sum.insert({item.first, sum});
363 }
364 // Containes the current index of agents located at each location
365 std::map<DiscretePoint, std::size_t> local_counts;
366
367 // Built agents
368 std::vector<api::model::GridAgent<CellType>*> agents;
369
370 this->build_agents(model, groups, [&agents, &factory] {
371 auto agent = factory();
372 // Keep a reference to built agents
373 agents.push_back((api::model::GridAgent<CellType>*) agent);
374 return agent;
375 }, agent_mapping);
376
377 agent_begin = GridAgentIndex::begin(item_counts);
378 agent_end = GridAgentIndex::end(item_counts);
379 std::vector<std::mt19937_64::result_type> local_seeds(
380 GridAgentIndex::distance(agent_begin, agent_end)
381 );
382 for(std::size_t i = 0; i < local_seeds.size(); i++)
383 // Generates a unique integer for each, independently from the
384 // current distribution
385 local_seeds[i] = i;
386 // Genererates a better seed distribution. The same seeds are
387 // generated on all processes.
388 std::seed_seq seed_generator(local_seeds.begin(), local_seeds.end());
389 seed_generator.generate(local_seeds.begin(), local_seeds.end());
390
391 for(std::size_t i = 0; i < agents.size(); i++) {
392 std::size_t& offset = local_counts[agents[i]->locationPoint()];
393 GridAgentIndex index(item_counts, agents[i]->locationPoint(), offset);
394 agent_index.insert({index, agents[i]});
395 agents[i]->seed(
396 local_seeds[GridAgentIndex::distance(agent_begin, index)]
397 );
398
399 ++offset;
400 }
401 }
