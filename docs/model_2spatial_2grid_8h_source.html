138 void GridAgent<AgentType, GridCellType, Derived>::moveTo(DiscretePoint point) {
139 bool found = false;
140 auto mobility_field = this->template outNeighbors<GridCellType>(SpatialModelLayers::MOVE);
141 auto it = mobility_field.begin();
142 while(!found && it != mobility_field.end()) {
143 if((*it)->location() == point) {
144 found=true;
145 } else {
146 it++;
147 }
148 }
149 if(found)
150 this->moveTo(*it);
151 else
152 throw api::model::OutOfMobilityFieldException(this->node()->getId(), this->locationPoint(), point);
153 }
