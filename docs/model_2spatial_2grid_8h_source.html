164 void GridAgent<AgentType, GridCellType, Derived>::moveTo(DiscretePoint point) {
165 bool found = false;
166 auto mobility_field = this->template outNeighbors<GridCellType>(SpatialModelLayers::MOVE);
167 auto it = mobility_field.begin();
168 while(!found && it != mobility_field.end()) {
169 if((*it)->location() == point) {
170 found=true;
171 } else {
172 it++;
173 }
174 }
175 if(found)
176 this->moveTo(*it);
177 else
178 throw api::model::OutOfMobilityFieldException(this->node()->getId(), this->locationPoint(), point);

409 template<typename CellType>
410 void GridAgentBuilder<CellType>::build(
411 api::model::SpatialModel<CellType>& model,
412 api::model::GroupList groups,
413 std::function<api::model::SpatialAgent<CellType>*()> factory,
414 api::model::SpatialAgentMapping<api::model::GridCell>& agent_mapping
415 ) {
416 for(auto cell : model.cells()) {
417 std::size_t count = agent_mapping.countAt(cell);
418 if(count > 0)
419 // Does not insert useless null entries
420 item_counts[cell->location()]=count;
421 }
422
423 // Gather item counts from all processes
424 communication::TypedMpi<decltype(item_counts)> item_counts_mpi(
425 model.getMpiCommunicator());
426 item_counts =
427 communication::all_reduce(item_counts_mpi, item_counts, utils::Concat());
428
429 // Contains the current index of agents located at each location
430 std::map<DiscretePoint, std::size_t> local_counts;
431
432 // Built agents
433 std::vector<api::model::GridAgent<CellType>*> agents;
434
435 this->build_agents(model, groups, [&agents, &factory] {
436 auto agent = factory();
437 // Keep a reference to built agents
438 agents.push_back((api::model::GridAgent<CellType>*) agent);
439 return agent;
440 }, agent_mapping);
441
442 agent_begin = GridAgentIndex::begin(&item_counts);
443 agent_end = GridAgentIndex::end(&item_counts);
444 std::vector<std::mt19937_64::result_type> local_seeds(
445 GridAgentIndex::distance(agent_begin, agent_end)
446 );
447 for(std::size_t i = 0; i < local_seeds.size(); i++)
448 // Generates a unique integer for each, independently from the
449 // current distribution
450 local_seeds[i] = i;
451 // Genererates a better seed distribution. The same seeds are
452 // generated on all processes.
453 std::seed_seq seed_generator(local_seeds.begin(), local_seeds.end());
454 seed_generator.generate(local_seeds.begin(), local_seeds.end());
455
456 for(std::size_t i = 0; i < agents.size(); i++) {
457 std::size_t& offset = local_counts[agents[i]->locationPoint()];
458 GridAgentIndex index(&item_counts, agents[i]->locationPoint(), offset);
459 agent_index.insert({index, agents[i]});
460
461 ++offset;
462 }
463 initSequence(local_seeds, [] (
464 api::model::GridAgent<CellType>* agent,
