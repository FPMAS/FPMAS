149 bool found = false;
150 auto mobility_field = this->template outNeighbors<GridCellType>(SpatialModelLayers::MOVE);
151 auto it = mobility_field.begin();
152 while(!found && it != mobility_field.end()) {
153 if((*it)->location() == point) {
154 found=true;
155 } else {
156 it++;
157 }
158 }
159 if(found)
160 this->moveTo(*it);
161 else
162 throw api::model::OutOfMobilityFieldException(this->node()->getId(), this->locationPoint(), point);
