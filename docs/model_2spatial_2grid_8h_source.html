139 void GridAgent<AgentType, GridCellType, Derived>::moveTo(DiscretePoint point) {
140 bool found = false;
141 auto mobility_field = this->template outNeighbors<GridCellType>(SpatialModelLayers::MOVE);
142 auto it = mobility_field.begin();
143 while(!found && it != mobility_field.end()) {
144 if((*it)->location() == point) {
145 found=true;
146 } else {
147 it++;
148 }
149 }
150 if(found)
151 this->moveTo(*it);
152 else
153 throw api::model::OutOfMobilityFieldException(this->node()->getId(), this->locationPoint(), point);
154 }
