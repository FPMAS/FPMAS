133  template<typename T>
134  void LinkClient<T>::link(const EdgeApi* edge) {
135  bool distantSrc = edge->getSourceNode()->state() == LocationState::DISTANT;
136  bool distantTgt = edge->getTargetNode()->state() == LocationState::DISTANT;
137 
138  if(edge->getSourceNode()->getLocation() != edge->getTargetNode()->getLocation()) {
139  // The edge is DISTANT, so at least of the two nodes is
140  // DISTANT. In this case, if the two nodes are
141  // DISTANT, they don't have the same location, so two
142  // requests must be performed.
143  MPI_Request reqSrc;
144  MPI_Request reqTgt;
145  // Simultaneously initiate the two requests, that are potentially
146  // made to different procs
147  if(distantSrc) {
148  edge_mpi.Issend(
149  const_cast<EdgeApi*>(edge), edge->getSourceNode()->getLocation(),
150  server_pack.getEpoch() | Tag::LINK, &reqSrc
151  );
152  }
153  if(distantTgt) {
154  edge_mpi.Issend(
155  const_cast<EdgeApi*>(edge), edge->getTargetNode()->getLocation(),
156  server_pack.getEpoch() | Tag::LINK, &reqTgt
157  );
158  }
159  // Sequentially waits for each request : if reqTgt completes
160  // before reqSrc, the second waitSendRequest will immediatly return
161  // so there is no performance issue.
162  if(distantSrc) {
163  server_pack.waitSendRequest(&reqSrc);
164  }
165  if(distantTgt) {
166  server_pack.waitSendRequest(&reqTgt);
167  }
168  } else {
169  // The edge is DISTANT, and its source and target nodes
170  // locations are the same, so the two nodes are necessarily
171  // located on the same DISTANT proc : only need to perform
172  // one request
173  MPI_Request req;
174  edge_mpi.Issend(
175  const_cast<EdgeApi*>(edge), edge->getSourceNode()->getLocation(),
176  server_pack.getEpoch() | Tag::LINK, &req
177  );
178  server_pack.waitSendRequest(&req);
179  }
180  }
