84  template<typename T>
85  void LinkClient<T>::link(const EdgeApi* edge) {
86  if(edge->state() == LocationState::DISTANT) {
87  bool distantSrc = edge->getSourceNode()->state() == LocationState::DISTANT;
88  bool distantTgt = edge->getTargetNode()->state() == LocationState::DISTANT;
89 
90  if(edge->getSourceNode()->getLocation() != edge->getTargetNode()->getLocation()) {
91  // The edge is DISTANT, so at least of the two nodes is
92  // DISTANT. In this case, if the two nodes are
93  // DISTANT, they don't have the same location, so two
94  // requests must be performed.
95  MPI_Request reqSrc;
96  MPI_Request reqTgt;
97  // Simultaneously initiate the two requests, that are potentially
98  // made to different procs
99  if(distantSrc) {
100  edge_mpi.Issend(
101  const_cast<EdgeApi*>(edge), edge->getSourceNode()->getLocation(),
102  server_pack.getEpoch() | Tag::LINK, &reqSrc
103  );
104  }
105  if(distantTgt) {
106  edge_mpi.Issend(
107  const_cast<EdgeApi*>(edge), edge->getTargetNode()->getLocation(),
108  server_pack.getEpoch() | Tag::LINK, &reqTgt
109  );
110  }
111  // Sequentially waits for each request : if reqTgt completes
112  // before reqSrc, the second waitSendRequest will immediatly return
113  // so there is no performance issue.
114  if(distantSrc) {
115  server_pack.waitSendRequest(&reqSrc);
116  }
117  if(distantTgt) {
118  server_pack.waitSendRequest(&reqTgt);
119  }
120  } else {
121  // The edge is DISTANT, and its source and target nodes
122  // locations are the same, so the two nodes are necessarily
123  // located on the same DISTANT proc : only need to perform
124  // one request
125  MPI_Request req;
126  edge_mpi.Issend(
127  const_cast<EdgeApi*>(edge), edge->getSourceNode()->getLocation(),
128  server_pack.getEpoch() | Tag::LINK, &req
129  );
130  server_pack.waitSendRequest(&req);
131  }
132  }
133  }
