74  template<typename T>
75  void LinkServer<T>::handleIncomingRequests() {
76  fpmas::api::communication::Status status;
77  // Check read request
78  if(edge_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::LINK, status)) {
79  EdgeApi* edge = edge_mpi.recv(status.source, status.tag);
80  FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive link request from %i", status.source);
81  graph.importEdge(edge);
82  }
83  if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::UNLINK, status)) {
84  DistributedId unlink_id = id_mpi.recv(status.source, status.tag);
85  FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive unlink request %s from %i", FPMAS_C_STR(unlink_id), status.source);
86  if(erased_edges_from_remove_node.count(unlink_id) == 0 && graph.getEdges().count(unlink_id) > 0) {
87  auto* edge = graph.getEdge(unlink_id);
88  graph.erase(graph.getEdge(unlink_id));
89  erased_edges_from_unlink.insert(edge);
90  }
91  }
92  if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::REMOVE_NODE, status)) {
93  DistributedId node_id = id_mpi.recv(status.source, status.tag);
94  FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive remove node request %s from %i", FPMAS_C_STR(node_id), status.source);
95  auto* node = graph.getNode(node_id);
96  // Unlinks all edges from the process which own the node to
97  // remove, because it's the only process that has an access to
98  // all the links
99  // Performs distant calls if required.
100  //
101  // When calling graph.unlink(), UNLINK request from other
102  // procs to unlink edges contained in node->getOutgoingEdges or
103  // node->getIncomingEdges might be received (for example, if an
104  // other process is removing a neighbor of `node`). In this
105  // case, the received UNLINK request has priority and the edge is added
106  // to the erase_edges_from_unlink set, and is ignore in the remove node
107  // process below.
108  // An other edge case needs to be avoid : while
109  // graph.unlink(edge) is processed, it is possible that an UNLINK
110  // request for edge is received. In this case, the remove node
111  // operation has priority, the edge is is added to
112  // erased_edges_from_remove_node, and the corresponding edge is
113  // ignored when the UNLINK request is processed.
114  //
115  // The two erased_edges sets are cleared once the node has been
116  // completely unlinked.
117  for(auto edge : node->getOutgoingEdges()) {
118  if(erased_edges_from_unlink.count(edge) == 0) {
119  erased_edges_from_remove_node.insert(edge->getId());
120  graph.unlink(edge);
121  }
122  }
123  for(auto edge : node->getIncomingEdges()) {
124  if(erased_edges_from_unlink.count(edge) == 0) {
125  erased_edges_from_remove_node.insert(edge->getId());
126  graph.unlink(edge);
127  }
128  }
129  erased_edges_from_unlink.clear();
130  erased_edges_from_remove_node.clear();
131 
132  // When this point is reached, it is guaranteed that the node
133  // is completely unlinked in the global graph, so no new
134  // requests to the node will be performed. However, it is
135  // possible that requests were pending when the node removal
136  // process was initialized. In consequence, to allow those
137  // request to finish, the node is not erased yet, but will be
138  // when HardSyncLinker::synchronize() is called, when it is sure
139  // that no more data request is ongoing.
140  sync_linker.registerNodeToRemove(node);
141  }
142  }

194  template<typename T>
195  void LinkClient<T>::link(const EdgeApi* edge) {
196  bool distant_src = edge->getSourceNode()->state() == LocationState::DISTANT;
197  bool distant_tgt = edge->getTargetNode()->state() == LocationState::DISTANT;
198 
199  if(edge->getSourceNode()->location() != edge->getTargetNode()->location()) {
200  // The edge is DISTANT, so at least of the two nodes is
201  // DISTANT. In this case, if the two nodes are
202  // DISTANT, they don't have the same location, so two
203  // requests must be performed.
204  fpmas::api::communication::Request req_src;
205  fpmas::api::communication::Request req_tgt;
206  // Simultaneously initiate the two requests, that are potentially
207  // made to different procs
208  if(distant_src) {
209  edge_mpi.Issend(
210  const_cast<EdgeApi*>(edge), edge->getSourceNode()->location(),
211  server_pack.getEpoch() | Tag::LINK, req_src
212  );
213  }
214  if(distant_tgt) {
215  edge_mpi.Issend(
216  const_cast<EdgeApi*>(edge), edge->getTargetNode()->location(),
217  server_pack.getEpoch() | Tag::LINK, req_tgt
218  );
219  }
220  // Sequentially waits for each request : if req_tgt completes
221  // before req_src, the second waitSendRequest will immediatly return
222  // so there is no performance issue.
223  if(distant_src) {
224  server_pack.waitSendRequest(req_src);
225  }
226  if(distant_tgt) {
227  server_pack.waitSendRequest(req_tgt);
228  }
229  } else {
230  // The edge is DISTANT, and its source and target nodes
231  // locations are the same, so the two nodes are necessarily
232  // located on the same DISTANT proc : only need to perform
233  // one request
234  fpmas::api::communication::Request req;
235  edge_mpi.Issend(
236  const_cast<EdgeApi*>(edge), edge->getSourceNode()->location(),
237  server_pack.getEpoch() | Tag::LINK, req
238  );
239  server_pack.waitSendRequest(req);
240  }
241  }
