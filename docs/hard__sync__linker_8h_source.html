98 template<typename T>
99 void LinkServer<T>::handleIncomingRequests() {
100 fpmas::api::communication::Status status;
101 // Check read request
102 if(edge_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::LINK, status)) {
103 EdgeApi* edge = edge_mpi.recv(status.source, status.tag);
104 FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive link request from %i", status.source);
105 graph.importEdge(edge);
106 }
107 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::UNLINK, status)) {
108 DistributedId unlink_id = id_mpi.recv(status.source, status.tag);
109 FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive unlink request %s from %i", FPMAS_C_STR(unlink_id), status.source);
110 if(
111 // The edge is not being unlinking by the local process
112 !isLockedUnlink(unlink_id)
113 // The edge has not been unlinked by an other UNLINK
114 // operation
115 && graph.getEdges().count(unlink_id) > 0) {
116 auto* edge = graph.getEdge(unlink_id);
117 // Source or target node is not being removed by the local
118 // process. In this case, the local process is responsible
119 // for all the required unlink operations, so the incoming
120 // request is ignored.
121 if(!(isLockedRemoveNode(edge->getSourceNode()->getId()) || isLockedRemoveNode(edge->getTargetNode()->getId()))) {
122 graph.erase(edge);
123 }
124 }
125 }
126 if(id_mpi.Iprobe(MPI_ANY_SOURCE, epoch | Tag::REMOVE_NODE, status)) {
127 DistributedId node_id = id_mpi.recv(status.source, status.tag);
128 FPMAS_LOGD(this->comm.getRank(), "LINK_SERVER", "receive remove node request %s from %i", FPMAS_C_STR(node_id), status.source);
129
130 // Initiates a removeNode operation from the local process,
131 // that will trigger all required UNLINK operations
132 graph.removeNode(graph.getNode(node_id));
133 }
134 }

186 template<typename T>
187 void LinkClient<T>::link(const EdgeApi* edge) {
188 bool distant_src = edge->getSourceNode()->state() == LocationState::DISTANT;
189 bool distant_tgt = edge->getTargetNode()->state() == LocationState::DISTANT;
190
191 if(edge->getSourceNode()->location() != edge->getTargetNode()->location()) {
192 // The edge is DISTANT, so at least of the two nodes is
193 // DISTANT. In this case, if the two nodes are
194 // DISTANT, they don't have the same location, so two
195 // requests must be performed.
196 fpmas::api::communication::Request req_src;
197 fpmas::api::communication::Request req_tgt;
198 // Simultaneously initiate the two requests, that are potentially
199 // made to different procs
200 if(distant_src) {
201 edge_mpi.Issend(
202 const_cast<EdgeApi*>(edge), edge->getSourceNode()->location(),
203 server_pack.getEpoch() | Tag::LINK, req_src
204 );
205 }
206 if(distant_tgt) {
207 edge_mpi.Issend(
208 const_cast<EdgeApi*>(edge), edge->getTargetNode()->location(),
209 server_pack.getEpoch() | Tag::LINK, req_tgt
210 );
211 }
212 // Sequentially waits for each request : if req_tgt completes
213 // before req_src, the second waitSendRequest will immediatly return
214 // so there is no performance issue.
215 if(distant_src) {
216 server_pack.waitSendRequest(req_src);
217 }
218 if(distant_tgt) {
219 server_pack.waitSendRequest(req_tgt);
220 }
221 } else {
222 // The edge is DISTANT, and its source and target nodes
223 // locations are the same, so the two nodes are necessarily
224 // located on the same DISTANT proc : only need to perform
225 // one request
226 fpmas::api::communication::Request req;
227 edge_mpi.Issend(
228 const_cast<EdgeApi*>(edge), edge->getSourceNode()->location(),
229 server_pack.getEpoch() | Tag::LINK, req
230 );
231 server_pack.waitSendRequest(req);
232 }
233 }

331 void link(EdgeApi* edge) override {
332 if(edge->state() == LocationState::DISTANT) {
333 link_client.link(edge);
334
335 if(edge->getSourceNode()->state() == LocationState::DISTANT
336 && edge->getTargetNode()->state() == LocationState::DISTANT) {
337 ghost_edges.push_back(const_cast<EdgeApi*>(edge));
338 }
339 }

351 void unlink(EdgeApi* edge) override {
352 // Prevents other processes to unlink the edge while the local
353 // process is unlinking it. In this case, incoming unlink requests
354 // have no effect. (see LinkServer::handleIncomingRequests())
355 server_pack.linkServer().lockUnlink(edge->getId());
356
357 if(edge->state() == LocationState::DISTANT) {
358 link_client.unlink(edge);
359 }
360
361 // Unlocks the unlink operation. The edge will actually be erased
362 // from the graph upon return, in the DistributedGraph::unlink()
363 // method.
364 server_pack.linkServer().unlockUnlink(edge->getId());
365 };

382 void removeNode(NodeApi* node) override {
383 if(node->state() == LocationState::DISTANT) {
384 link_client.removeNode(node);
385 } else {
386 server_pack.linkServer().lockRemoveNode(node->getId());
387 for(auto edge : node->getOutgoingEdges())
388 graph.unlink(edge);
389 for(auto edge : node->getIncomingEdges())
390 graph.unlink(edge);
391 server_pack.linkServer().unlockRemoveNode(node->getId());
392 }
393 registerNodeToRemove(node);
394 }
395
419 void synchronize() override {
420 FPMAS_LOGI(graph.getMpiCommunicator().getRank(), "HARD_SYNC_LINKER", "Synchronizing sync linker...", "");
421 for(auto edge : ghost_edges)
422 graph.erase(edge);
423 ghost_edges.clear();
424
425 server_pack.terminate();
426
427 for(auto node : nodes_to_remove) {
428 graph.erase(node);
429 }
430 nodes_to_remove.clear();
431
432 FPMAS_LOGI(graph.getMpiCommunicator().getRank(), "HARD_SYNC_LINKER", "Synchronized.", "");
433 };
