327 typename GridBuilder<CellType>::CellMatrix GridBuilder<CellType>::buildLocalCells(
328 api::model::SpatialModel<CellType>& model,
329 GridDimensions local_dimensions,
330 api::model::GroupList groups) const {
331
332 DiscretePoint origin = local_dimensions.getOrigin();
333 DiscretePoint extents = local_dimensions.getExtent();
334
335 CellMatrix cells;
336 allocate(cells, local_dimensions.width(), local_dimensions.height());
337
338 for(DiscreteCoordinate y = origin.y; y < extents.y; y++) {
339 for(DiscreteCoordinate x = origin.x; x < extents.x; x++) {
340 auto cell = cell_factory.build({x, y});
341 cells[y-origin.y][x-origin.x] = cell;
342 model.add(cell);
343 for(auto group : groups)
344 group.get().add(cell);
345 }
346 }
347 return cells;
