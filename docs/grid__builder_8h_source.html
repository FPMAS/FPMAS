432 template<typename CellType>
433 typename GridBuilder<CellType>::CellMatrix GridBuilder<CellType>::buildLocalCells(
434 api::model::SpatialModel<CellType>& model,
435 GridDimensions local_dimensions,
436 api::model::GroupList groups) const {
437
438 DiscretePoint origin = local_dimensions.getOrigin();
439 DiscretePoint extents = local_dimensions.getExtent();
440
441 CellMatrix cells;
442 allocate(cells, local_dimensions.width(), local_dimensions.height());
443
444 for(DiscreteCoordinate y = origin.y; y < extents.y; y++) {
445 for(DiscreteCoordinate x = origin.x; x < extents.x; x++) {
446 auto cell = cell_factory.build({x, y});
447 cells[y-origin.y][x-origin.x] = cell;
448 model.add(cell);
449 for(auto group : groups)
450 group.get().add(cell);
451 }
452 }
453
454 for(DiscreteCoordinate y = 0; y < this->height(); y++)
455 for(DiscreteCoordinate x = 0; x < this->width(); x++)
456 built_cells.insert(std::pair<api::model::DiscretePoint, std::size_t>(
457 {x, y}, 1ul
458 ));
459
460 cell_begin = GridCellIndex::begin(&built_cells);
461 cell_end = GridCellIndex::end(&built_cells);
462
463 for(auto row : cells) {
464 for(auto cell : row) {
465 // GridCellIndex offset is always 0 in this case (at most one
466 // cell per coordinate)
467 GridCellIndex index(&built_cells, cell->location(), 0);
468 cell_index.insert({index, cell});
469 }
470 }
471
472 // Random cell seed initialization
473 std::vector<std::mt19937_64::result_type> local_seeds(
474 GridCellIndex::distance(cell_begin, cell_end)
475 );
476 for(std::size_t i = 0; i < local_seeds.size(); i++)
477 // Generates a unique integer for each, independently from the
478 // current distribution
479 local_seeds[i] = i;
480 // Genererates a better seed distribution. The same seeds are
481 // generated on all processes.
482 std::seed_seq seed_generator(local_seeds.begin(), local_seeds.end());
483 seed_generator.generate(local_seeds.begin(), local_seeds.end());
484
485 initSequence(local_seeds, [] (
486 CellType* cell,
487 const std::mt19937_64::result_type& seed) {
488 cell->seed(seed);
489 });
490
491 return cells;
492 }
