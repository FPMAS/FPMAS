344 // sync_mode have been constructed in `graph` using
345 // `graph.mpi_communicator`. Since we reassign
346 // `this->mpi_communicator` to `graph.mpi_communicator`, we can

348 // without introducing inconsistencies.
349 this->mpi_communicator = graph.mpi_communicator;
350 this->id_mpi = std::move(graph.id_mpi);
351 this->location_mpi = std::move(graph.location_mpi);
352 this->node_mpi = std::move(graph.node_mpi);
353 this->edge_mpi = std::move(graph.edge_mpi);
354
355 this->location_manager = std::move(graph.location_manager);
356 this->sync_mode = std::move(graph.sync_mode);
357 // The obsolete `this->erase_node_callback` is
358 // automatically deleted by the base Graph move assignment,
359 // when the "erase_node_callback" list is cleared.
360 this->erase_node_callback = graph.erase_node_callback;
361 // Prevents erase_node_callback->disable() when `graph` is
362 // deleted
363 graph.erase_node_callback = nullptr;
364
365 this->node_id = std::move(graph.node_id);
366 this->edge_id = std::move(graph.edge_id);

407 template<DIST_GRAPH_PARAMS>
408 typename DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::NodeType*
409 DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::importNode(NodeType* node) {
410 FPMAS_LOGD(getMpiCommunicator().getRank(), "DIST_GRAPH", "Importing node %s...", FPMAS_C_STR(node->getId()));
411 // The input node must be a temporary dynamically allocated object.
412 // A representation of the node might already be contained in the
413 // graph, if it were already built as a "distant" node.
414
415 if(this->getNodes().count(node->getId())==0) {
416 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Inserting new LOCAL node %s.", FPMAS_C_STR(node->getId()));
417 // The node is not contained in the graph, we need to build a new
418 // one.
419 // But instead of completely building a new node, we can re-use the
420 // temporary input node.
421 this->insert(node);
422 setLocal(node);
423 node->setMutex(sync_mode.buildMutex(node));
424 return node;
425 }
426 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Replacing existing DISTANT node %s.", FPMAS_C_STR(node->getId()));
427 // A representation of the node was already contained in the graph.
428 // We just need to update its state.
429
430 // Set local representation as local
431 auto local_node = this->getNode(node->getId());
432 local_node->data() = std::move(node->data());
433 local_node->setWeight(node->getWeight());
434 setLocal(local_node);
435
436 // Deletes unused temporary input node
437 delete node;
438
439 return local_node;
440 }

442 template<DIST_GRAPH_PARAMS>
443 typename DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::EdgeType*
444 DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::importEdge(EdgeType* edge) {
445 FPMAS_LOGD(getMpiCommunicator().getRank(), "DIST_GRAPH", "Importing edge %s (from %s to %s)...",
446 FPMAS_C_STR(edge->getId()),
447 FPMAS_C_STR(edge->getSourceNode()->getId()),
448 FPMAS_C_STR(edge->getTargetNode()->getId())
449 );
450 // The input edge must be a dynamically allocated object, with temporary
451 // dynamically allocated nodes as source and target.
452 // A representation of the imported edge might already be present in the
453 // graph, for example if it has already been imported as a "distant"
454 // edge with other nodes at other epochs.
455
456 if(this->getEdges().count(edge->getId())==0) {
457 // The edge does not belong to the graph : a new one must be built.
458
459 DistributedId src_id = edge->getSourceNode()->getId();
460 NodeType* src;
461 DistributedId tgt_id = edge->getTargetNode()->getId();
462 NodeType* tgt;
463
464 LocationState edgeLocationState = LocationState::LOCAL;
465
466 if(this->getNodes().count(src_id) > 0) {
467 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing source %s", FPMAS_C_STR(src_id));
468 // The source node is already contained in the graph
469 src = this->getNode(src_id);
470 if(src->state() == LocationState::DISTANT) {
471 // At least src is DISTANT, so the imported edge is
472 // necessarily DISTANT.
473 edgeLocationState = LocationState::DISTANT;
474 }
475 // Deletes the temporary source node
476 delete edge->getSourceNode();
477
478 // Links the temporary edge with the src contained in the graph
479 edge->setSourceNode(src);
480 src->linkOut(edge);
481 } else {
482 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT source %s", FPMAS_C_STR(src_id));
483 // The source node is not contained in the graph : it must be
484 // built as a DISTANT node.
485 edgeLocationState = LocationState::DISTANT;
486
487 // Instead of building a new node, we re-use the temporary
488 // source node.
489 src = edge->getSourceNode();
490 this->insert(src);
491 setDistant(src);
492 src->setMutex(sync_mode.buildMutex(src));
493 unsynchronized_nodes.insert(src);
494 }
495 if(this->getNodes().count(tgt_id) > 0) {
496 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing target %s", FPMAS_C_STR(tgt_id));
497 // The target node is already contained in the graph
498 tgt = this->getNode(tgt_id);
499 if(tgt->state() == LocationState::DISTANT) {
500 // At least src is DISTANT, so the imported edge is
501 // necessarily DISTANT.
502 edgeLocationState = LocationState::DISTANT;
503 }
504 // Deletes the temporary target node
505 delete edge->getTargetNode();
506
507 // Links the temporary edge with the tgt contained in the graph
508 edge->setTargetNode(tgt);
509 tgt->linkIn(edge);
510 } else {
511 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT target %s", FPMAS_C_STR(tgt_id));
512 // The target node is not contained in the graph : it must be
513 // built as a DISTANT node.
514 edgeLocationState = LocationState::DISTANT;
515
516 // Instead of building a new node, we re-use the temporary
517 // target node.
518 tgt = edge->getTargetNode();
519 this->insert(tgt);
520 setDistant(tgt);
521 tgt->setMutex(sync_mode.buildMutex(tgt));
522 unsynchronized_nodes.insert(tgt);
523 }
524 // Finally, insert the temporary edge into the graph.
525 edge->setState(edgeLocationState);
526 this->insert(edge);
527 return edge;
528 } // if (graph.count(edge_id) > 0)
529
530 // A representation of the edge is already present in the graph : it is
531 // useless to insert it again. We just need to update its state.
532
533 auto local_edge = this->getEdge(edge->getId());
534 if(local_edge->getSourceNode()->state() == LocationState::LOCAL
535 && local_edge->getTargetNode()->state() == LocationState::LOCAL) {
536 local_edge->setState(LocationState::LOCAL);
537 }
538
539 // Completely deletes temporary items, nothing is re-used
540 delete edge->getSourceNode();
541 delete edge->getTargetNode();
542 delete edge;
543
544 return local_edge;
545 }

634 template<DIST_GRAPH_PARAMS>
635 void DistributedGraph<DIST_GRAPH_PARAMS_SPEC>
636 ::_distribute(api::graph::PartitionMap partition) {
637 FPMAS_LOGI(getMpiCommunicator().getRank(), "DIST_GRAPH",
638 "Distributing graph...", "");
639 std::string partition_str = "\n";
640 for(auto item : partition) {
641 std::string str = FPMAS_C_STR(item.first);
642 str.append(" : " + std::to_string(item.second) + "\n");
643 partition_str.append(str);
644 }
645 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Partition : %s", partition_str.c_str());
646
647
648 // Builds node and edges export maps
649 std::vector<NodeType*> exported_nodes;
650 std::unordered_map<int, std::vector<NodePtrWrapper<T>>> node_export_map;
651 std::unordered_map<int, std::set<DistributedId>> edge_ids_to_export;
652 std::unordered_map<int, std::vector<EdgePtrWrapper<T>>> edge_export_map;
653 for(auto item : partition) {
654 if(this->getNodes().count(item.first) > 0) {
655 if(item.second != mpi_communicator->getRank()) {
656 FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH",
657 "Exporting node %s to %i", FPMAS_C_STR(item.first), item.second);
658 auto node_to_export = this->getNode(item.first);
659 exported_nodes.push_back(node_to_export);
660 node_export_map[item.second].emplace_back(node_to_export);
661 for(auto edge : node_to_export->getIncomingEdges()) {
662 // Insert or replace in the IDs set
663 edge_ids_to_export[item.second].insert(edge->getId());
664 }
665 for(auto edge : node_to_export->getOutgoingEdges()) {
666 // Insert or replace in the IDs set
667 edge_ids_to_export[item.second].insert(edge->getId());
668 }
669 }
670 }
671 }
672 // Ensures that each edge is exported once to each process
673 for(auto list : edge_ids_to_export) {
674 for(auto id : list.second) {
675 edge_export_map[list.first].emplace_back(this->getEdge(id));
676 }
677 }
678
679 {
680 // Serialize and export / import nodes
681 auto node_import = node_mpi.migrate(node_export_map);
682 for(auto& import_node_list_from_proc : node_import) {
683 for(auto& imported_node : import_node_list_from_proc.second) {
684 this->importNode(imported_node);
685 }
686 }
687 // node_import is deleted at the end of this scope
688 }
689
690 {
691 // Serialize and export / import edges
692 auto edge_import = edge_mpi.migrate(edge_export_map);
693 for(auto& import_edge_list_from_proc : edge_import) {
694 for(auto& imported_edge : import_edge_list_from_proc.second) {
695 this->importEdge(imported_edge);
696 }
697 }
698 // edge_import is deleted at the end of this scope

765 ::clearNode(NodeType* node) {
766 DistributedId id = node->getId();
767 FPMAS_LOGD(
768 mpi_communicator->getRank(),
769 "DIST_GRAPH", "Clearing node %s...",
770 FPMAS_C_STR(id)
771 );
772
773 bool eraseNode = true;
774 std::set<EdgeType*> obsoleteEdges;
775 for(auto edge : node->getIncomingEdges()) {
776 if(edge->getSourceNode()->state()==LocationState::LOCAL) {
777 eraseNode = false;
778 } else {
779 obsoleteEdges.insert(edge);
780 }
781 }
782 for(auto edge : node->getOutgoingEdges()) {
783 if(edge->getTargetNode()->state()==LocationState::LOCAL) {
784 eraseNode = false;
785 } else {
786 obsoleteEdges.insert(edge);
787 }
788 }
789 if(eraseNode) {
790 FPMAS_LOGD(
791 mpi_communicator->getRank(),
792 "DIST_GRAPH", "Erasing obsolete node %s.",
793 FPMAS_C_STR(node->getId())
794 );
795 this->erase(node);
796 } else {
797 for(auto edge : obsoleteEdges) {
798 FPMAS_LOGD(
799 mpi_communicator->getRank(),
800 "DIST_GRAPH", "Erasing obsolete edge %s (%p) (from %s to %s)",
801 FPMAS_C_STR(node->getId()), edge,
802 FPMAS_C_STR(edge->getSourceNode()->getId()),
803 FPMAS_C_STR(edge->getTargetNode()->getId())
804 );
805 this->erase(edge);
806 }
807 }
808 FPMAS_LOGD(
809 mpi_communicator->getRank(),
810 "DIST_GRAPH", "Node %s cleared.",
811 FPMAS_C_STR(id)
812 );
813 }
