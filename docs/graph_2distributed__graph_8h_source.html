324  // dynamically allocated nodes as source and target.
325  // A representation of the imported edge might already be present in the
326  // graph, for example if it has already been imported as a "distant"
327  // edge with other nodes at other epochs.
328 
329  if(this->getEdges().count(edge->getId())==0) {
330  // The edge does not belong to the graph : a new one must be built.
331 
332  DistributedId srcId = edge->getSourceNode()->getId();
333  NodeType* src;
334  DistributedId tgtId = edge->getTargetNode()->getId();
335  NodeType* tgt;
336 
337  LocationState edgeLocationState = LocationState::LOCAL;
338 
339  if(this->getNodes().count(srcId) > 0) {
340  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing source %s", FPMAS_C_STR(srcId));
341  // The source node is already contained in the graph
342  src = this->getNode(srcId);
343  if(src->state() == LocationState::DISTANT) {
344  // At least src is DISTANT, so the imported edge is
345  // necessarily DISTANT.
346  edgeLocationState = LocationState::DISTANT;
347  }
348  // Deletes the temporary source node
349  delete edge->getSourceNode();
350 
351  // Links the temporary edge with the src contained in the graph
352  edge->setSourceNode(src);
353  src->linkOut(edge);
354  } else {
355  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT source %s", FPMAS_C_STR(srcId));
356  // The source node is not contained in the graph : it must be
357  // built as a DISTANT node.
358  edgeLocationState = LocationState::DISTANT;
359 
360  // Instead of building a new node, we re-use the temporary
361  // source node.
362  src = edge->getSourceNode();
363  this->insert(src);
364  setDistant(src);
365  src->setMutex(sync_mode.buildMutex(src));
366  }
367  if(this->getNodes().count(tgtId) > 0) {
368  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing target %s", FPMAS_C_STR(tgtId));
369  // The target node is already contained in the graph
370  tgt = this->getNode(tgtId);
371  if(tgt->state() == LocationState::DISTANT) {
372  // At least src is DISTANT, so the imported edge is
373  // necessarily DISTANT.
374  edgeLocationState = LocationState::DISTANT;
375  }
376  // Deletes the temporary target node
377  delete edge->getTargetNode();
378 
379  // Links the temporary edge with the tgt contained in the graph
380  edge->setTargetNode(tgt);
381  tgt->linkIn(edge);
382  } else {
383  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT target %s", FPMAS_C_STR(tgtId));
384  // The target node is not contained in the graph : it must be
385  // built as a DISTANT node.
386  edgeLocationState = LocationState::DISTANT;
387 
388  // Instead of building a new node, we re-use the temporary
389  // target node.
390  tgt = edge->getTargetNode();
391  this->insert(tgt);
392  setDistant(tgt);
393  tgt->setMutex(sync_mode.buildMutex(tgt));
394  }
395  // Finally, insert the temporary edge into the graph.
396  edge->setState(edgeLocationState);
397  this->insert(edge);
398  return edge;
399  } // if (graph.count(edge_id) > 0)
400 
401  // A representation of the edge is already present in the graph : it is
402  // useless to insert it again. We just need to update its state.
403 
404  auto local_edge = this->getEdge(edge->getId());

407  local_edge->setState(LocationState::LOCAL);
408  }
409

411  delete edge->getSourceNode();
412  delete edge->getTargetNode();

494  "Distributing graph...", "");
495  std::string partition_str = "\n";
496  for(auto item : partition) {
497  std::string str = FPMAS_C_STR(item.first);
498  str.append(" : " + std::to_string(item.second) + "\n");
499  partition_str.append(str);
500  }
501  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Partition : %s", partition_str.c_str());
502 
503  sync_mode.getSyncLinker().synchronize();
504 
505  // Builds node and edges export maps
506  std::vector<NodeType*> exported_nodes;
507  std::unordered_map<int, std::vector<NodePtrWrapper<T>>> node_export_map;
508  std::unordered_map<int, std::set<DistributedId>> edge_ids_to_export;
509  std::unordered_map<int, std::vector<EdgePtrWrapper<T>>> edge_export_map;
510  for(auto item : partition) {
511  if(this->getNodes().count(item.first) > 0) {
512  if(item.second != mpi_communicator.getRank()) {
513  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH",
514  "Exporting node %s to %i", FPMAS_C_STR(item.first), item.second);
515  auto node_to_export = this->getNode(item.first);
516  exported_nodes.push_back(node_to_export);
517  node_export_map[item.second].emplace_back(node_to_export);
518  for(auto edge : node_to_export->getIncomingEdges()) {
519  // Insert or replace in the IDs set
520  edge_ids_to_export[item.second].insert(edge->getId());
521  }
522  for(auto edge : node_to_export->getOutgoingEdges()) {
523  // Insert or replace in the IDs set
524  edge_ids_to_export[item.second].insert(edge->getId());
525  }
526  }
527  }
528  }
529  // Ensures that each edge is exported once to each process
530  for(auto list : edge_ids_to_export) {
531  for(auto id : list.second) {
532  edge_export_map[list.first].emplace_back(this->getEdge(id));
533  }
534  }
535 
536  // Serialize and export / import nodes
537  auto node_import = node_mpi.migrate(node_export_map);
538 
539  // Serialize and export / import edges
540  auto edge_import = edge_mpi.migrate(edge_export_map);
541 
542  for(auto& import_node_list_from_proc : node_import) {
543  for(auto& imported_node : import_node_list_from_proc.second) {
544  this->importNode(imported_node);
545  }
546  }
547  for(auto& import_edge_list_from_proc : edge_import) {
548  for(auto& imported_edge : import_edge_list_from_proc.second) {
549  this->importEdge(imported_edge);
550  }
551  }
552

554  setDistant(node);
555  }

557  location_manager.updateLocations();
558 
559  FPMAS_LOGD(getMpiCommunicator().getRank(), "DIST_GRAPH", "Clearing exported nodes...", "");
560  for(auto node : exported_nodes) {

562  }

564
