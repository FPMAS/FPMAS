363  DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::importNode(NodeType* node) {
364  FPMAS_LOGD(getMpiCommunicator().getRank(), "DIST_GRAPH", "Importing node %s...", FPMAS_C_STR(node->getId()));
365  // The input node must be a temporary dynamically allocated object.
366  // A representation of the node might already be contained in the
367  // graph, if it were already built as a "distant" node.
368 
369  if(this->getNodes().count(node->getId())==0) {
370  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Inserting new LOCAL node %s.", FPMAS_C_STR(node->getId()));
371  // The node is not contained in the graph, we need to build a new
372  // one.
373  // But instead of completely building a new node, we can re-use the
374  // temporary input node.
375  this->insert(node);
376  setLocal(node);
377  node->setMutex(sync_mode.buildMutex(node));
378  return node;
379  }
380  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Replacing existing DISTANT node %s.", FPMAS_C_STR(node->getId()));
381  // A representation of the node was already contained in the graph.
382  // We just need to update its state.
383 
384  // Set local representation as local
385  auto local_node = this->getNode(node->getId());
386  local_node->data() = std::move(node->data());
387  local_node->setWeight(node->getWeight());
388  setLocal(local_node);
389 
390  // Deletes unused temporary input node
391  delete node;
392 
393  return local_node;
394  }

397  typename DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::EdgeType*
398  DistributedGraph<DIST_GRAPH_PARAMS_SPEC>::importEdge(EdgeType* edge) {
399  FPMAS_LOGD(getMpiCommunicator().getRank(), "DIST_GRAPH", "Importing edge %s (from %s to %s)...",
400  FPMAS_C_STR(edge->getId()),
401  FPMAS_C_STR(edge->getSourceNode()->getId()),
402  FPMAS_C_STR(edge->getTargetNode()->getId())
403  );
404  // The input edge must be a dynamically allocated object, with temporary
405  // dynamically allocated nodes as source and target.
406  // A representation of the imported edge might already be present in the
407  // graph, for example if it has already been imported as a "distant"
408  // edge with other nodes at other epochs.
409 
410  if(this->getEdges().count(edge->getId())==0) {
411  // The edge does not belong to the graph : a new one must be built.
412 
413  DistributedId src_id = edge->getSourceNode()->getId();
414  NodeType* src;
415  DistributedId tgt_id = edge->getTargetNode()->getId();
416  NodeType* tgt;
417 
418  LocationState edgeLocationState = LocationState::LOCAL;
419 
420  if(this->getNodes().count(src_id) > 0) {
421  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing source %s", FPMAS_C_STR(src_id));
422  // The source node is already contained in the graph
423  src = this->getNode(src_id);
424  if(src->state() == LocationState::DISTANT) {
425  // At least src is DISTANT, so the imported edge is
426  // necessarily DISTANT.
427  edgeLocationState = LocationState::DISTANT;
428  }
429  // Deletes the temporary source node
430  delete edge->getSourceNode();
431 
432  // Links the temporary edge with the src contained in the graph
433  edge->setSourceNode(src);
434  src->linkOut(edge);
435  } else {
436  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT source %s", FPMAS_C_STR(src_id));
437  // The source node is not contained in the graph : it must be
438  // built as a DISTANT node.
439  edgeLocationState = LocationState::DISTANT;
440 
441  // Instead of building a new node, we re-use the temporary
442  // source node.
443  src = edge->getSourceNode();
444  this->insert(src);
445  setDistant(src);
446  src->setMutex(sync_mode.buildMutex(src));
447  }
448  if(this->getNodes().count(tgt_id) > 0) {
449  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Linking existing target %s", FPMAS_C_STR(tgt_id));
450  // The target node is already contained in the graph
451  tgt = this->getNode(tgt_id);
452  if(tgt->state() == LocationState::DISTANT) {
453  // At least src is DISTANT, so the imported edge is
454  // necessarily DISTANT.
455  edgeLocationState = LocationState::DISTANT;
456  }
457  // Deletes the temporary target node
458  delete edge->getTargetNode();
459 
460  // Links the temporary edge with the tgt contained in the graph
461  edge->setTargetNode(tgt);
462  tgt->linkIn(edge);
463  } else {
464  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Creating DISTANT target %s", FPMAS_C_STR(tgt_id));
465  // The target node is not contained in the graph : it must be
466  // built as a DISTANT node.
467  edgeLocationState = LocationState::DISTANT;
468 
469  // Instead of building a new node, we re-use the temporary
470  // target node.
471  tgt = edge->getTargetNode();
472  this->insert(tgt);
473  setDistant(tgt);
474  tgt->setMutex(sync_mode.buildMutex(tgt));
475  }
476  // Finally, insert the temporary edge into the graph.
477  edge->setState(edgeLocationState);
478  this->insert(edge);
479  return edge;
480  } // if (graph.count(edge_id) > 0)
481 
482  // A representation of the edge is already present in the graph : it is
483  // useless to insert it again. We just need to update its state.
484 
485  auto local_edge = this->getEdge(edge->getId());
486  if(local_edge->getSourceNode()->state() == LocationState::LOCAL
487  && local_edge->getTargetNode()->state() == LocationState::LOCAL) {
488  local_edge->setState(LocationState::LOCAL);
489  }
490 
491  // Completely deletes temporary items, nothing is re-used
492  delete edge->getSourceNode();
493  delete edge->getTargetNode();
494  delete edge;
495 
496  return local_edge;
497  }

599  ::_distribute(api::graph::PartitionMap partition) {
600  FPMAS_LOGI(getMpiCommunicator().getRank(), "DIST_GRAPH",
601  "Distributing graph...", "");
602  std::string partition_str = "\n";
603  for(auto item : partition) {
604  std::string str = FPMAS_C_STR(item.first);
605  str.append(" : " + std::to_string(item.second) + "\n");
606  partition_str.append(str);
607  }
608  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH", "Partition : %s", partition_str.c_str());
609 
610 
611  // Builds node and edges export maps
612  std::vector<NodeType*> exported_nodes;
613  std::unordered_map<int, std::vector<NodePtrWrapper<T>>> node_export_map;
614  std::unordered_map<int, std::set<DistributedId>> edge_ids_to_export;
615  std::unordered_map<int, std::vector<EdgePtrWrapper<T>>> edge_export_map;
616  for(auto item : partition) {
617  if(this->getNodes().count(item.first) > 0) {
618  if(item.second != mpi_communicator->getRank()) {
619  FPMAS_LOGV(getMpiCommunicator().getRank(), "DIST_GRAPH",
620  "Exporting node %s to %i", FPMAS_C_STR(item.first), item.second);
621  auto node_to_export = this->getNode(item.first);
622  exported_nodes.push_back(node_to_export);
623  node_export_map[item.second].emplace_back(node_to_export);
624  for(auto edge : node_to_export->getIncomingEdges()) {
625  // Insert or replace in the IDs set
626  edge_ids_to_export[item.second].insert(edge->getId());
627  }
628  for(auto edge : node_to_export->getOutgoingEdges()) {
629  // Insert or replace in the IDs set
630  edge_ids_to_export[item.second].insert(edge->getId());
631  }
632  }
633  }
634  }
635  // Ensures that each edge is exported once to each process
636  for(auto list : edge_ids_to_export) {
637  for(auto id : list.second) {
638  edge_export_map[list.first].emplace_back(this->getEdge(id));
639  }
640  }
641 
642  // Serialize and export / import nodes
643  auto node_import = node_mpi.migrate(node_export_map);
644 
645  // Serialize and export / import edges
646  auto edge_import = edge_mpi.migrate(edge_export_map);
647 
648  for(auto& import_node_list_from_proc : node_import) {
649  for(auto& imported_node : import_node_list_from_proc.second) {
650  this->importNode(imported_node);
651  }
652  }
