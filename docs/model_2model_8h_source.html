675 // The purpose of "moveAssign" is clearly inconsistent with its
676 // current behavior, that does much more that just "moving" the
677 // agent.

679 // Sets and overrides the fields that must be preserved
680 std::set<api::model::GroupId> local_ids;
681 for(auto id : this->groupIds())
682 local_ids.insert(id);
683 std::set<api::model::GroupId> updated_ids;
684 for(auto id : agent->groupIds())
685 updated_ids.insert(id);
686
687 std::vector<api::model::GroupId> new_groups;
688 for(auto id : updated_ids)
689 if(local_ids.count(id) == 0)
690 new_groups.push_back(id);
691
692 std::vector<api::model::GroupId> obsolete_groups;
693 for(auto id : local_ids)
694 if(updated_ids.count(id) == 0)
695 obsolete_groups.push_back(id);
696
697 // Preserve the local agent set up
698 for(auto group : this->groups())
699 agent->addGroup(group);
700 // Ok because the updated list of ids is saved in updated_ids
701 for(auto id : agent->groupIds())
702 agent->removeGroupId(id);
703 for(auto id : this->groupIds())
704 agent->addGroupId(id);
705
706 for(auto task : this->tasks())
707 agent->setTask(task.first, task.second);
708 agent->setNode(this->node());
709 agent->setModel(this->model());
710
711 // Uses AgentType move assignment operator
712 //
713 // groupIds(), groups(), tasks(), node() and model() are
714 // notably moved from agent to this, but this as no effect
715 // since agent's fields was overriden above. In consequence,
716 // those fields are properly preserved, as required by the
717 // method.
718 *static_cast<AgentType*>(this) = std::move(*static_cast<AgentType*>(agent));
719
720 // Dynamically updates groups lists
721 // If `agent` was added to / removed from group on a distant
722 // process for example (assuming that this agent is DISTANT),
723 // this agent representation groups are updated.
724 for(auto id : new_groups)
725 this->model()->getGroup(id).add(this);
726 for(auto id : obsolete_groups)
727 this->model()->getGroup(id).remove(this);
728 }
