684  // current behavior, that does much more that just "moving" the
685  // agent.

687  // Sets and overrides the fields that must be preserved
688  std::set<api::model::GroupId> local_ids;
689  for(auto id : this->groupIds())
690  local_ids.insert(id);
691  std::set<api::model::GroupId> updated_ids;
692  for(auto id : agent->groupIds())
693  updated_ids.insert(id);
694 
695  std::vector<api::model::GroupId> new_groups;
696  for(auto id : updated_ids)
697  if(local_ids.count(id) == 0)
698  new_groups.push_back(id);
699 
700  std::vector<api::model::GroupId> obsolete_groups;
701  for(auto id : local_ids)
702  if(updated_ids.count(id) == 0)
703  obsolete_groups.push_back(id);
704 
705  // Preserve the local agent set up
706  for(auto group : this->groups())
707  agent->addGroup(group);
708  // Ok because the updated list of ids is saves in updated_ids
709  for(auto id : agent->groupIds())
710  agent->removeGroupId(id);
711  for(auto id : this->groupIds())
712  agent->addGroupId(id);
713 
714  for(auto task : this->tasks())
715  agent->setTask(task.first, task.second);
716  agent->setNode(this->node());
717  agent->setModel(this->model());
718 
719  // Uses AgentType move assignment operator
720  //
721  // groupIds(), groups(), tasks(), node() and model() are
722  // notably moved from agent to this, but this as no effect
723  // since agent's fields was overriden above. In consequence,
724  // those fields are properly preserved, as required by the
725  // method.
726  *dynamic_cast<AgentType*>(this) = std::move(*dynamic_cast<AgentType*>(agent));
