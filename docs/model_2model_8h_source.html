693 void moveAssign(api::model::Agent* agent) override {
694 // TODO: this should probably be improved in 2.0
695 // The purpose of "moveAssign" is clearly inconsistent with its
696 // current behavior, that does much more that just "moving" the
697 // agent.
698
699 // Sets and overrides the fields that must be preserved
700 std::set<api::model::GroupId> local_ids;
701 for(auto id : this->groupIds())
702 local_ids.insert(id);
703 std::set<api::model::GroupId> updated_ids;
704 for(auto id : agent->groupIds())
705 updated_ids.insert(id);
706
707 std::vector<api::model::GroupId> new_groups;
708 for(auto id : updated_ids)
709 if(local_ids.count(id) == 0)
710 new_groups.push_back(id);
711
712 std::vector<api::model::GroupId> obsolete_groups;
713 for(auto id : local_ids)
714 if(updated_ids.count(id) == 0)
715 obsolete_groups.push_back(id);
716
717 // Preserve the local agent set up
718 for(auto group : this->groups())
719 agent->addGroup(group);
720 // Ok because the updated list of ids is saved in updated_ids
721 for(auto id : agent->groupIds())
722 agent->removeGroupId(id);
723 for(auto id : this->groupIds())
724 agent->addGroupId(id);
725
726 for(auto task : this->tasks())
727 agent->setTask(task.first, task.second);
728 agent->setNode(this->node());
729 agent->setModel(this->model());
730
731 // Uses AgentType move assignment operator
732 //
733 // groupIds(), groups(), tasks(), node() and model() are
734 // notably moved from agent to this, but this as no effect
735 // since agent's fields was overriden above. In consequence,
736 // those fields are properly preserved, as required by the
737 // method.
738 *dynamic_cast<AgentType*>(this) = std::move(*dynamic_cast<AgentType*>(agent));
739
740 // Dynamically updates groups lists
741 // If `agent` was added to / removed from group on a distant
742 // process for example (assuming that this agent is DISTANT),
743 // this agent representation groups are updated.
744 for(auto id : new_groups)
745 this->model()->getGroup(id).add(this);
746 for(auto id : obsolete_groups)
747 this->model()->getGroup(id).remove(this);
748 }
