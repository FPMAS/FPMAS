682 void moveAssign(api::model::Agent* agent) override {
683 // TODO: this should probably be improved in 2.0
684 // The purpose of "moveAssign" is clearly inconsistent with its
685 // current behavior, that does much more that just "moving" the
686 // agent.
687
688 // Sets and overrides the fields that must be preserved
689 std::set<api::model::GroupId> local_ids;
690 for(auto id : this->groupIds())
691 local_ids.insert(id);
692 std::set<api::model::GroupId> updated_ids;
693 for(auto id : agent->groupIds())
694 updated_ids.insert(id);
695
696 std::vector<api::model::GroupId> new_groups;
697 for(auto id : updated_ids)
698 if(local_ids.count(id) == 0)
699 new_groups.push_back(id);
700
701 std::vector<api::model::GroupId> obsolete_groups;
702 for(auto id : local_ids)
703 if(updated_ids.count(id) == 0)
704 obsolete_groups.push_back(id);
705
706 // Preserve the local agent set up
707 for(auto group : this->groups())
708 agent->addGroup(group);
709 // Ok because the updated list of ids is saved in updated_ids
710 for(auto id : agent->groupIds())
711 agent->removeGroupId(id);
712 for(auto id : this->groupIds())
713 agent->addGroupId(id);
714
715 for(auto task : this->tasks())
716 agent->setTask(task.first, task.second);
717 agent->setNode(this->node());
718 agent->setModel(this->model());
719
720 // Uses AgentType move assignment operator
721 //
722 // groupIds(), groups(), tasks(), node() and model() are
723 // notably moved from agent to this, but this as no effect
724 // since agent's fields was overriden above. In consequence,
725 // those fields are properly preserved, as required by the
726 // method.
727 *dynamic_cast<AgentType*>(this) = std::move(*dynamic_cast<AgentType*>(agent));
728
729 // Dynamically updates groups lists
730 // If `agent` was added to / removed from group on a distant
731 // process for example (assuming that this agent is DISTANT),
732 // this agent representation groups are updated.
733 for(auto id : new_groups)
734 this->model()->getGroup(id).add(this);
735 for(auto id : obsolete_groups)
736 this->model()->getGroup(id).remove(this);
737 }
