625  void moveAssign(api::model::Agent* agent) override {
626  // TODO: this should probably be improved in 2.0
627  // The purpose of "moveAssign" is clearly inconsistent with its
628  // current behavior, that does much more that just "moving" the
629  // agent.
630 
631  // Sets and overrides the fields that must be preserved
632  std::set<api::model::GroupId> local_ids;
633  for(auto id : this->groupIds())
634  local_ids.insert(id);
635  std::set<api::model::GroupId> updated_ids;
636  for(auto id : agent->groupIds())
637  updated_ids.insert(id);
638 
639  std::vector<api::model::GroupId> new_groups;
640  for(auto id : updated_ids)
641  if(local_ids.count(id) == 0)
642  new_groups.push_back(id);
643 
644  std::vector<api::model::GroupId> obsolete_groups;
645  for(auto id : local_ids)
646  if(updated_ids.count(id) == 0)
647  obsolete_groups.push_back(id);
648 
649  // Preserve the local agent set up
650  for(auto group : this->groups())
651  agent->addGroup(group);
652  // Ok because the updated list of ids is saves in updated_ids
653  for(auto id : agent->groupIds())
654  agent->removeGroupId(id);
655  for(auto id : this->groupIds())
656  agent->addGroupId(id);
657 
658  for(auto task : this->tasks())
659  agent->setTask(task.first, task.second);
660  agent->setNode(this->node());
661  agent->setModel(this->model());
662 
663  // Uses AgentType move assignment operator
664  //
665  // groupIds(), groups(), tasks(), node() and model() are
666  // notably moved from agent to this, but this as no effect
667  // since agent's fields was overriden above. In consequence,
668  // those fields are properly preserved, as required by the

670  *dynamic_cast<AgentType*>(this) = std::move(*dynamic_cast<AgentType*>(agent));
