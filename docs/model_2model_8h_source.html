781 void moveAssign(api::model::Agent* agent) override {
782 // TODO: this should probably be improved in 2.0
783 // The purpose of "moveAssign" is clearly inconsistent with its
784 // current behavior, that does much more that just "moving" the
785 // agent.
786
787 // Sets and overrides the fields that must be preserved
788 std::set<api::model::GroupId> local_ids;
789 for(auto id : this->groupIds())
790 local_ids.insert(id);
791 std::set<api::model::GroupId> updated_ids;
792 for(auto id : agent->groupIds())
793 updated_ids.insert(id);
794
795 std::vector<api::model::GroupId> new_groups;
796 for(auto id : updated_ids)
797 if(local_ids.count(id) == 0)
798 new_groups.push_back(id);
799
800 std::vector<api::model::GroupId> obsolete_groups;
801 for(auto id : local_ids)
802 if(updated_ids.count(id) == 0)
803 obsolete_groups.push_back(id);
804
805 // Uses AgentType move assignment operator
806 //
807 // groupIds(), groups(), tasks(), node() and model() are
808 // notably moved from agent to this, but this as no effect
809 // considering the move assignment operator defined above. In
810 // consequence, those fields are properly preserved, as
811 // required by the method.
812 // If AgentType only explicitly defines a copy assignment
813 // operator, it should not call the AgentBase copy assignment
814 // so there should not be any problem.
815 *static_cast<AgentType*>(this) = std::move(*static_cast<AgentType*>(agent));
816
817 // Dynamically updates groups lists
818 // If `agent` was added to / removed from group on a distant
819 // process for example (assuming that this agent is DISTANT),
820 // this agent representation groups are updated.
821 for(auto id : new_groups)
822 this->model()->getGroup(id).add(this);
823 for(auto id : obsolete_groups)
824 this->model()->getGroup(id).remove(this);
825 }
