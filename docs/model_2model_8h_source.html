911 void moveAssign(api::model::Agent* agent) override {

914 // current behavior, that does much more that just "moving" the
915 // agent.

917 // Sets and overrides the fields that must be preserved
918 std::set<api::model::GroupId> local_ids;
919 for(auto id : this->groupIds())
920 local_ids.insert(id);
921 std::set<api::model::GroupId> updated_ids;
922 for(auto id : agent->groupIds())
923 updated_ids.insert(id);
924
925 std::vector<api::model::GroupId> new_groups;
926 for(auto id : updated_ids)
927 if(local_ids.count(id) == 0)
928 new_groups.push_back(id);
929
930 std::vector<api::model::GroupId> obsolete_groups;
931 for(auto id : local_ids)
932 if(updated_ids.count(id) == 0)
933 obsolete_groups.push_back(id);
934
935 // Uses AgentType move assignment operator
936 //
937 // groupIds(), groups(), tasks(), node() and model() are
938 // notably moved from agent to this, but this as no effect
939 // considering the move assignment operator defined above. In
940 // consequence, those fields are properly preserved, as
941 // required by the method.
942 // If AgentType only explicitly defines a move assignment
943 // operator, it should not call the AgentBase move assignment
944 // so there should not be any problem.
945 *static_cast<AgentType*>(this) = std::move(*static_cast<AgentType*>(agent));
946
947 // Dynamically updates groups lists
948 // If `agent` was added to / removed from group on a distant
949 // process for example (assuming that this agent is DISTANT),
950 // this agent representation groups are updated.
951 for(auto id : new_groups)
952 this->model()->getGroup(id).add(this);
953 for(auto id : obsolete_groups)
