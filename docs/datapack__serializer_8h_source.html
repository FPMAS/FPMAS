660 static std::size_t size(const PackType& p, const WeakAgentPtr& ptr) {
661 if(ptr->typeId() == Type::TYPE_ID) {
662 return p.template size<api::model::TypeId>()
663 + p.template size(TypedAgentPtr<Type>(
664 const_cast<Type*>(dynamic_cast<const Type*>(ptr.get()))
665 ))
666 + p.template size(ptr->groupIds());
667 }
668 return AgentPtrSerializer<PackType, AgentTypes...>::size(p, ptr);
669 }

688 static void to_datapack(
689 PackType& p, const WeakAgentPtr& ptr) {
690 if(ptr->typeId() == Type::TYPE_ID) {
691 p.put(Type::TYPE_ID);
692 p.put(TypedAgentPtr<Type>(
693 const_cast<Type*>(dynamic_cast<const Type*>(ptr.get()))
694 ));
695 p.put(ptr->groupIds());
696 } else {
697 AgentPtrSerializer<PackType, AgentTypes...>::to_datapack(p, ptr);
698 }
699 }

718 static WeakAgentPtr from_datapack(const PackType& p) {
719 std::size_t pos = p.readOffset();
720 fpmas::api::model::TypeId id = p
721 .template get<fpmas::api::model::TypeId>();
722 if(id == Type::TYPE_ID) {
723 auto agent = p
724 .template get<TypedAgentPtr<Type>>();
725 for(auto gid :
726 p.template get<std::vector<api::model::GroupId>>())
727 agent->addGroupId(gid);
728 return {agent};
729 } else {
730 p.seekRead(pos);
731 return AgentPtrSerializer<PackType, AgentTypes...>::from_datapack(p);
732 }
733 }
