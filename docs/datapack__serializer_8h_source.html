522 if(ptr->typeId() == Type::TYPE_ID) {
523 PackType type_id = Type::TYPE_ID;

531 );
532 p.write(type_id);

557 static WeakAgentPtr from_datapack(const PackType& p) {
558 std::size_t pos = p.readPos();
559 fpmas::api::model::TypeId id = p
560 .template read<PackType>()
561 .template get<fpmas::api::model::TypeId>();
562 if(id == Type::TYPE_ID) {
