160  void GhostSyncLinker<T>::synchronize() {
161  /*
162  * Migrate links
163  */
164  std::vector<EdgePtr> edges_to_clear;
165  std::unordered_map<int, std::vector<EdgePtr>> link_migration;
166  for(auto edge : link_buffer) {
167  auto src = edge->getSourceNode();
168  if(src->state() == LocationState::DISTANT) {
169  link_migration[src->getLocation()].push_back(edge);
170  }
171  auto tgt = edge->getTargetNode();
172  if(tgt->state() == LocationState::DISTANT) {
173  link_migration[tgt->getLocation()].push_back(edge);
174  }
175  if(src->state() == LocationState::DISTANT
176  && tgt->state() == LocationState::DISTANT) {
177  edges_to_clear.push_back(edge);
178  }
179  }
180  link_migration = edge_mpi.migrate(link_migration);
181 
182  for(auto import_list : link_migration) {
183  for (auto edge : import_list.second) {
184  graph.importEdge(edge);
185  }
186  }
187  link_buffer.clear();
188 
189  /*
190  * Migrate unlinks
191  */
192  unlink_migration = id_mpi.migrate(unlink_migration);
193  for(auto import_list : unlink_migration) {
194  for(DistributedId id : import_list.second) {
195  if(graph.getEdges().count(id) > 0) {
196  auto edge = graph.getEdge(id);
197  graph.erase(edge);
198  }
199  }
200  }
201  unlink_migration.clear();
202  for(auto edge : edges_to_clear)
203  graph.erase(edge);
204  }
