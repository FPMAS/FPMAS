334 void GhostSyncLinker<T>::synchronize() {
335 FPMAS_LOGI(graph.getMpiCommunicator().getRank(), "GHOST_MODE", "Synchronizing graph links...", "");
336 /*
337 * Migrate links
338 */
339 std::vector<EdgePtr> edges_to_clear;
340 std::unordered_map<int, std::vector<EdgePtr>> link_migration;
341 for(auto edge : link_buffer) {
342 auto src = edge->getSourceNode();
343 if(src->state() == LocationState::DISTANT) {
344 link_migration[src->location()].push_back(edge);
345 }
346 auto tgt = edge->getTargetNode();
347 if(tgt->state() == LocationState::DISTANT) {
348 link_migration[tgt->location()].push_back(edge);
349 }
350 if(src->state() == LocationState::DISTANT
351 && tgt->state() == LocationState::DISTANT) {
352 edges_to_clear.push_back(edge);
353 }
354 }
355 link_migration = edge_mpi.migrate(link_migration);
356
357 for(auto import_list : link_migration) {
358 for (auto edge : import_list.second) {
359 graph.importEdge(edge);
360 }
361 }
362 link_buffer.clear();
363
364 /*
365 * Migrate node removal
366 */
367 remove_node_buffer = id_mpi.migrate(remove_node_buffer);
368 for(auto import_list : remove_node_buffer) {
369 for(DistributedId node_id : import_list.second) {
370 auto* node = graph.getNode(node_id);
371 for(auto edge : node->getOutgoingEdges())
372 graph.unlink(edge);
373 for(auto edge : node->getIncomingEdges())
374 graph.unlink(edge);
375 }
376 }
377
378 /*
379 * Migrate unlinks
380 */
381 unlink_migration = id_mpi.migrate(unlink_migration);
382 for(auto import_list : unlink_migration) {
383 for(DistributedId id : import_list.second) {
384 if(graph.getEdges().count(id) > 0) {
385 auto edge = graph.getEdge(id);
386 graph.erase(edge);
387 }
388 }
389 }
390 unlink_migration.clear();
391
392 for(auto edge : edges_to_clear)
393 graph.erase(edge);
394
395 for(auto import_list : remove_node_buffer)
396 for(auto node_id : import_list.second)
397 local_nodes_to_remove.push_back(graph.getNode(node_id));
398 remove_node_buffer.clear();
399
400 for(auto node : local_nodes_to_remove) {
401 graph.erase(node);
402 }
403 local_nodes_to_remove.clear();
